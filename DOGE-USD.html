def calculate_rsi(data, period=14): ""Calculates the Relative Strength Index (RSI)."" delta = data[Close].diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0

avg_gain = up.rolling(window=period, min_periods=period).mean()
avg_loss = abs(down.rolling(window=period, min_periods=period).mean())

rs = avg_gain / avg_loss
rsi = 100 - (100 / (1 + rs))
return rsi
