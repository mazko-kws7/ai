Function to calculate Relative Strength Index (RSI)

def calculate_rsi(data, period=14): delta = data[Close].diff() up, down = delta.clip(lower=0), -1*delta.clip(upper=0) roll_up1 = up.ewm(alpha=1/period).mean() roll_down1 = down.ewm(alpha=1/period).mean() RS = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS)) return RSI
