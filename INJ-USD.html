import numpy as np

def calculate_rsi(close_prices, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    deltas = np.diff(close_prices)
    seed = deltas[:period+1]
    up = seed[seed >= 0].sum()/period
    down = -seed[seed < 0].sum()/period
    rs = up/down
    rsi = np.zeros_like(close_prices)
    rsi[:period] = 100. - 100./(1.+rs)

    for i in range(period, len(close_prices)):
        delta = deltas[i-1] # the diff is 1 shorter

        if delta > 0:
            upval = delta
            downval = 0.
        else:
            upval = 0.
            downval = -delta

        up = (up*(period-1) + upval)/period
        down = (down*(period-1) + downval)/period

        rs = up/down
        rsi[i] = 100. - 100./(1.+rs)

    return rsi

def calculate_moving_average(data, period):
    """Calculates the simple moving average."""
    return np.convolve(data, np.ones(period), 'valid') / period

def identify_support_resistance(data, lookback=10):
    """Identifies potential support and resistance levels based on local lows and highs."""
    support_levels = []
    resistance_levels = []

    for i in range(lookback, len(data) - lookback):
        is_support = True
        is_resistance = True

        for j in range(-lookback, lookback + 1):
            if data[i + j] < data[i]:
                is_support = False
            if data[i + j] > data[i]:
                is_resistance = False

        if is_support:
            support_levels.append((i, data[i]))
        if is_resistance:
            resistance_levels.append((i, data[i]))

    return support_levels, resistance_levels
