import numpy as np

def calculate_moving_average(data, period):
    """Calculates the moving average of a given dataset.

    Args:
        data (list): A list of numerical data.
        period (int): The period for the moving average.

    Returns:
        list: A list of moving averages.
    """
    if len(data) < period:
        return [None] * len(data)  # Not enough data points to calculate MA
    
    # Use numpy for efficient calculation
    np_data = np.array(data)
    moving_averages = np.convolve(np_data, np.ones(period), 'valid') / period
    
    # Pad the beginning with None values to match original data length
    padding = [None] * (period - 1)
    return padding + list(moving_averages)

def calculate_rsi(data, period):
    """Calculates the Relative Strength Index (RSI) of a given dataset.

    Args:
        data (list): A list of numerical data (e.g., closing prices).
        period (int): The period for the RSI calculation.

    Returns:
        list: A list of RSI values.
    """
    if len(data) < period:
        return [None] * len(data)

    # Calculate price changes
    deltas = np.diff(data)

    # Separate gains and losses
    seed = deltas[:period]
    up = seed[seed >= 0].sum() / period
    down = -seed[seed < 0].sum() / period

    rs = up / down if down != 0 else 0
    rsi = np.array([100. - 100. / (1. + rs)])

    # Calculate RSI for the rest of the data
    for delta in deltas[period:]:
        up = delta if delta > 0 else 0
        down = -delta if delta < 0 else 0

        up = (up + (period - 1) * up) / period
        down = (down + (period - 1) * down) / period

        rs = up / down if down != 0 else 0
        rsi = np.append(rsi, 100. - 100. / (1. + rs))

    # Pad the beginning with None values
    padding = [None] * period
    return padding + list(rsi)

def calculate_macd(data, fast_period, slow_period, signal_period):
    """Calculates the Moving Average Convergence Divergence (MACD) and signal line.

    Args:
        data (list): A list of numerical data (e.g., closing prices).
        fast_period (int): The period for the fast EMA.
        slow_period (int): The period for the slow EMA.
        signal_period (int): The period for the signal line EMA.

    Returns:
        tuple: A tuple containing two lists: MACD values and signal line values.
    """
    if len(data) < slow_period:
        return [None] * len(data), [None] * len(data)

    # Convert to numpy array for easier calculations
    data = np.array(data)

    # Calculate fast and slow EMAs
    ema_fast = calculate_ema(data, fast_period)
    ema_slow = calculate_ema(data, slow_period)

    # Calculate MACD
    macd = ema_fast[slow_period - 1:] - ema_slow[slow_period - 1:]

    # Calculate signal line
    signal = calculate_ema(macd, signal_period)

    # Pad the beginning with None values
    macd_padding = [None] * (slow_period - 1)
    signal_padding = [None] * (slow_period + signal_period - 2)  # Corrected padding for signal

    return macd_padding + list(macd), signal_padding + list(signal)


def calculate_ema(data, period):
    """Calculates the Exponential Moving Average (EMA) of a given dataset.

    Args:
        data (list): A list of numerical data.
        period (int): The period for the EMA calculation.

    Returns:
        list: A list of EMA values.
    """
    if len(data) < period:
        return [None] * len(data)

    # Convert to numpy array
    data = np.array(data)

    # Calculate initial SMA
    sma = np.mean(data[:period])

    # Calculate smoothing factor
    k = 2 / (period + 1)

    # Initialize EMA list with SMA
    ema = [sma]

    # Calculate EMA for the rest of the data
    for i in range(period, len(data)):
        ema.append(data[i] * k + ema[-1] * (1 - k))

    # Pad the beginning with None values
    padding = [None] * period
    return padding[:period-1] + list(ema)

def convert_to_float(data):
    """
    Converts a list of strings to a list of floats.
    Replaces invalid values with None.
    """
    float_data = []
    for x in data:
        try:
            float_data.append(float(x))
        except ValueError:
            float_data.append(None)
    return float_data

def identify_candlestick_patterns(open, high, low, close):
    """
    Identifies basic candlestick patterns.
    """
    patterns = []
    for i in range(len(close)):
        if open[i] is None or high[i] is None or low[i] is None or close[i] is None:
            patterns.append("No Pattern")
            continue

        if close[i] > open[i]:
            patterns.append("Bullish Candle")
        elif close[i] < open[i]:
            patterns.append("Bearish Candle")
        else:
            patterns.append("Doji")
    return patterns
