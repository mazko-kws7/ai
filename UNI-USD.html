import numpy as np
import pandas as pd

def calculate_rsi(data, window=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = data.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.rolling(window).mean()
    roll_down1 = down.abs().rolling(window).mean()
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

def calculate_sma(data, window=20):
    """Calculates the Simple Moving Average (SMA)."""
    return data.rolling(window).mean()

def analyze_ohlcv(ohlcv_data):
    """
    Performs technical analysis on OHLCV data.

    Args:
        ohlcv_data (pd.DataFrame): DataFrame with 'Open', 'High', 'Low', 'Close', 'Volume' columns.

    Returns:
        dict: Analysis results including SMA and RSI.
    """
    close_prices = ohlcv_data['Close']
    volume = ohlcv_data['Volume']
    sma_20 = calculate_sma(close_prices, window=20)
    rsi_14 = calculate_rsi(close_prices, window=14)
    volume_sma_10 = calculate_sma(volume, window=10)  # Volume SMA

    analysis = {
        'SMA_20': sma_20.iloc[-1],
        'RSI_14': rsi_14.iloc[-1],
        'Volume_SMA_10': volume_sma_10.iloc[-1]
    }
    return analysis
