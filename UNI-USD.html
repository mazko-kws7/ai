import numpy as np

def calculate_sma(data, period):
    if len(data) < period:
        return np.nan
    return np.mean(data[-period:])

def calculate_rsi(closes, period=14):
    if len(closes) <= period:
        return np.nan

    deltas = np.diff(closes)
    seed = deltas[:period+1]
    up = seed[seed >= 0].sum() / period
    down = -seed[seed < 0].sum() / period
    rs = up / down
    rsi = np.where(down == 0, 100, 100 - (100 / (1 + rs)))
    return rsi[-1]

def calculate_macd(closes, fast_period=12, slow_period=26, signal_period=9):
    if len(closes) <= slow_period:
        return np.nan, np.nan, np.nan

    fast_ema = calculate_ema(closes, fast_period)
    slow_ema = calculate_ema(closes, slow_period)
    macd_line = fast_ema - slow_ema
    signal_line = calculate_ema(np.array([macd_line]), signal_period) # Use array to make it compatible with calculate_ema
    histogram = macd_line - signal_line
    return macd_line, signal_line, histogram

def calculate_ema(data, period):
    if len(data) < period:
        return np.nan

    k = 2 / (period + 1)
    ema = np.zeros_like(data)
    ema[:period] = np.nan
    ema[period] = np.mean(data[:period])

    for i in range(period + 1, len(data)):
        ema[i] = (data[i] * k) + (ema[i-1] * (1 - k))
    return ema[-1]
