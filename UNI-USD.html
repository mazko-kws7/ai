import numpy as np

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = data.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = up.rolling(window=period, min_periods=period).mean()
    avg_loss = abs(down.rolling(window=period, min_periods=period).mean())
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def analyze_moving_averages(data, short_period=20, long_period=50):
    """Analyzes short and long term moving averages for crossovers."""
    short_ma = data.rolling(window=short_period, min_periods=short_period).mean()
    long_ma = data.rolling(window=long_period, min_periods=long_period).mean()
    return short_ma, long_ma

def assess_volume(volume_data):
    """Simple assessment of recent volume changes."""
    if len(volume_data) < 2:
        return "Neutral Volume"
    
    latest_volume = volume_data[-1]
    previous_volume = volume_data[-2]
    
    if latest_volume > previous_volume * 1.5:
        return "High Volume Increase"
    elif latest_volume < previous_volume * 0.5:
        return "Low Volume Decrease"
    else:
        return "Stable Volume"
