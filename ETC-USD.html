def calculate_rsi(prices, period=14):
    deltas = []
    for i in range(1, len(prices)):
        deltas.append(prices[i] - prices[i-1])
    
    if len(deltas) < period:
        return None  # Not enough data

    avg_gain = 0
    avg_loss = 0

    # First average gain and loss
    for i in range(period):
        if deltas[i] > 0:
            avg_gain += deltas[i]
        else:
            avg_loss += abs(deltas[i])
    avg_gain /= period
    avg_loss /= period

    rsi_values = [0] * len(prices)  # Initialize RSI array

    # Calculate RSI for remaining periods
    for i in range(period, len(prices) - 1):
      gain = deltas[i] if deltas[i] > 0 else 0
      loss = abs(deltas[i]) if deltas[i] < 0 else 0

      avg_gain = (avg_gain * (period - 1) + gain) / period
      avg_loss = (avg_loss * (period - 1) + loss) / period

      if avg_loss == 0:
          rsi = 100  # Avoid division by zero, max RSI
      else:
          rs = avg_gain / avg_loss
          rsi = 100 - (100 / (1 + rs))

      rsi_values[i + 1] = rsi

    return rsi_values

def calculate_sma(prices, period=20):
    if len(prices) < period:
        return None  # Not enough data

    sma_values = []
    for i in range(period, len(prices) + 1):
        window = prices[i-period:i]
        average = sum(window) / period
        sma_values.append(average)
    return sma_values

def calculate_macd(prices, fast_period=12, slow_period=26, signal_period=9):
    # Calculate Exponential Moving Averages (EMAs)
    def calculate_ema(prices, period):
        ema = [sum(prices[:period]) / period]  # Initialize with SMA
        k = 2 / (period + 1)
        for i in range(period, len(prices)):
            ema.append(prices[i] * k + ema[-1] * (1 - k))
        return ema

    ema_fast = calculate_ema(prices, fast_period)
    ema_slow = calculate_ema(prices, slow_period)

    # Calculate MACD Line
    macd_line = [ema_fast[i] - ema_slow[i] for i in range(len(ema_fast))]

    # Calculate Signal Line
    signal_line = calculate_ema(macd_line, signal_period)

    # Calculate Histogram (optional)
    # histogram = [macd_line[i] - signal_line[i] for i in range(len(signal_line))]

    return macd_line, signal_line
