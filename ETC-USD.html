def calculate_indicators(df): """ Calculates technical indicators for a given DataFrame.

Args:
    df (pd.DataFrame): DataFrame with 'Open', 'High', 'Low', 'Close', 'Volume' columns.

Returns:
    pd.DataFrame: DataFrame with added technical indicators.
"""
# Simple Moving Average (SMA) - 14 days
df['SMA_14'] = df['Close'].rolling(window=14).mean()

# Relative Strength Index (RSI) - 14 days
delta = df['Close'].diff()
gain = delta.where(delta > 0, 0)
loss = -delta.where(delta < 0, 0)
avg_gain = gain.rolling(window=14).mean()
avg_loss = loss.rolling(window=14).mean()
rs = avg_gain / avg_loss
df['RSI_14'] = 100 - (100 / (1 + rs))

# Moving Average Convergence Divergence (MACD)
exp12 = df['Close'].ewm(span=12, adjust=False).mean()
exp26 = df['Close'].ewm(span=26, adjust=False).mean()
df['MACD'] = exp12 - exp26
df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()

#Bollinger Bands (20 days, 2 std)
df['BB_MA'] = df['Close'].rolling(window=20).mean()
df['BB_Std'] = df['Close'].rolling(window=20).std()
df['BB_Upper'] = df['BB_MA'] + 2 * df['BB_Std']
df['BB_Lower'] = df['BB_MA'] - 2 * df['BB_Std']

return df
