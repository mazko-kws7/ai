df = pd.DataFrame(daily_data, columns=['Open', 'High', 'Low', 'Close', 'Volume'])

# Calculate 7-day SMA
df['SMA_7'] = df['Close'].rolling(window=7).mean()

# Calculate daily price changes
df['Price_Change'] = df['Close'].diff()

# Calculate gains and losses
df['Gain'] = df['Price_Change'].apply(lambda x: x if x > 0 else 0)
df['Loss'] = df['Price_Change'].apply(lambda x: abs(x) if x < 0 else 0)

# Calculate average gains and losses over 14 days
df['Avg_Gain_14'] = df['Gain'].rolling(window=14, min_periods=14).mean()
df['Avg_Loss_14'] = df['Loss'].rolling(window=14, min_periods=14).mean()

# Calculate RS
df['RS'] = df['Avg_Gain_14'] / df['Avg_Loss_14']

# Calculate RSI
df['RSI_14'] = 100 - (100 / (1 + df['RS']))
