# Calculate RSI (Relative Strength Index)
def calculate_rsi(data, period=14):
    delta = data['Close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.rolling(period).mean()
    roll_down1 = np.abs(down.rolling(period).mean())
    RS1 = roll_up1 / roll_down1
    RSI1 = 100.0 - (100.0 / (1.0 + RS1))
    return RSI1

df['RSI'] = calculate_rsi(df)

# Calculate MACD (Moving Average Convergence Divergence)
def calculate_macd(data, nslow=26, nfast=12):
    EMAfast = data['Close'].ewm(span=nfast, min_periods=nfast).mean()
    EMAslow = data['Close'].ewm(span=nslow, min_periods=nslow).mean()
    MACD = pd.Series(EMAfast - EMAslow, name='MACD')
    return MACD

df['MACD'] = calculate_macd(df)
