def calculate_sma(data, period): if len(data) < period: return [] sma_values = [] for i in range(period, len(data) + 1): window = data[i-period:i] sma = sum(window) / period sma_values.append(sma) return sma_values

def calculate_rsi(prices, period=14): if len(prices) <= period: return []

rsi_values = []
for i in range(period, len(prices)):
    changes = [prices[j] - prices[j - 1] for j in range(i - period + 1, i + 1)]
    gains = [chg for chg in changes if chg > 0]
    losses = [-chg for chg in changes if chg < 0]

    avg_gain = sum(gains) / period if gains else 0
    avg_loss = sum(losses) / period if losses else 0

    if avg_loss == 0:
        rsi = 100 if avg_gain > 0 else 50
    else:
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))

    rsi_values.append(rsi)
return rsi_values
