def calculate_rsi(data, period=14): delta = data[Close].diff(1) delta = delta.dropna() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 avg_up = up.rolling(window=period).mean() avg_down = abs(down.rolling(window=period).mean()) rs = avg_up / avg_down rsi = 100.0 - (100.0 / (1.0 + rs)) return rsi
