def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    import pandas as pd
    delta = pd.Series(data).diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.ewm(span=period, adjust=False).mean()
    roll_down1 = abs(down.ewm(span=period, adjust=False).mean())
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI.iloc[-1]

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    import pandas as pd
    ema_fast = pd.Series(data).ewm(span=fast_period, adjust=False).mean()
    ema_slow = pd.Series(data).ewm(span=slow_period, adjust=False).mean()
    macd = ema_fast - ema_slow
    signal = macd.ewm(span=signal_period, adjust=False).mean()
    histogram = macd - signal
    return macd.iloc[-1], signal.iloc[-1], histogram.iloc[-1]

def calculate_sma(data, period=20):
  """Calculates the Simple Moving Average (SMA)."""
  import pandas as pd
  sma = pd.Series(data).rolling(window=period).mean()
  return sma.iloc[-1]

def assess_trend(data, period=5):
    """Assess the trend over a given period. Returns 'up', 'down', or 'sideways'."""
    import pandas as pd
    closes = pd.Series(data)
    if closes.iloc[-1] > closes.iloc[-period]:
        return 'up'
    elif closes.iloc[-1] < closes.iloc[-period]:
        return 'down'
    else:
        return 'sideways'
