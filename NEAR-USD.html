# Calculate 50-day Moving Average
def calculate_ma(data, period=50):
    close_prices = [d[3] for d in data]
    if len(close_prices) < period:
        return None
    ma = sum(close_prices[-period:]) / period
    return ma

# Calculate RSI (Relative Strength Index)
def calculate_rsi(data, period=14):
    close_prices = [d[3] for d in data]
    if len(close_prices) <= period:
        return None
    
    # Calculate price changes
    price_changes = [close_prices[i] - close_prices[i-1] for i in range(1, len(close_prices))]
    
    # Get gains and losses
    gains = [change for change in price_changes[-period:] if change > 0]
    losses = [-change for change in price_changes[-period:] if change < 0]
    
    # Calculate average gain and loss
    average_gain = sum(gains) / period if gains else 0
    average_loss = sum(losses) / period if losses else 0
    
    # Calculate RSI
    if average_loss == 0:
        return 100
    rs = average_gain / average_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

# Calculate MACD (Moving Average Convergence Divergence)
def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    close_prices = [d[3] for d in data]
    if len(close_prices) <= slow_period:
        return None
    
    # Calculate EMA (Exponential Moving Average)
    def calculate_ema(prices, period):
        ema = [sum(prices[:period]) / period]
        k = 2 / (period + 1)
        for price in prices[period:]:
            ema.append(price * k + ema[-1] * (1 - k))
        return ema
    
    # Calculate fast and slow EMAs
    ema_fast = calculate_ema(close_prices[-slow_period:], fast_period)
    ema_slow = calculate_ema(close_prices[-slow_period:], slow_period)
    
    # Calculate MACD line
    macd_line = [ema_fast[i] - ema_slow[i] for i in range(len(ema_fast))]
    
    # Calculate signal line
    signal_line = calculate_ema(macd_line, signal_period)
    
    return macd_line[-1], signal_line[-1]
