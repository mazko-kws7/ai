def parse_data(data):
    lines = data.strip().split('\n')
    parsed_data = []
    for line in lines:
        values = line.split(',')
        open_price = float(values[0])
        high_price = float(values[1])
        low_price = float(values[2])
        close_price = float(values[3])
        volume = int(values[4])
        parsed_data.append({
            'Open': open_price,
            'High': high_price,
            'Low': low_price,
            'Close': close_price,
            'Volume': volume
        })
    return parsed_data
