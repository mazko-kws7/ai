import pandas as pd
import numpy as np

def calculate_sma(data, period):
    """Calculates the Simple Moving Average (SMA)."""
    return pd.Series(data).rolling(window=period).mean().fillna(method='bfill').to_list()

def calculate_rsi(close, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = pd.Series(close).diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    avg_gain = up.rolling(window=period, min_periods=period).mean()
    avg_loss = abs(down.rolling(window=period, min_periods=period).mean())

    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi.fillna(50).to_list()

def calculate_macd(close, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    close_series = pd.Series(close)
    ema_fast = close_series.ewm(span=fast_period, adjust=False).mean()
    ema_slow = close_series.ewm(span=slow_period, adjust=False).mean()
    macd_line = ema_fast - ema_slow
    signal_line = macd_line.ewm(span=signal_period, adjust=False).mean()
    histogram = macd_line - signal_line
    return macd_line.to_list(), signal_line.to_list(), histogram.to_list()
