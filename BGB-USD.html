import numpy as np

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    if len(data) < period:
        return np.nan
    return np.mean(data[-period:])

def calculate_rsi(data, period=14):
    """Calculates Relative Strength Index."""
    if len(data) < period:
        return np.nan

    deltas = np.diff(data)
    seed = deltas[:period]
    up = seed[seed >= 0].sum() / period
    down = -seed[seed < 0].sum() / period
    rs = up / down if down != 0 else 0  # Handle potential division by zero
    rsi = 100.0 - 100.0 / (1.0 + rs)
    return rsi

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """Calculates Moving Average Convergence Divergence (MACD)."""
    if len(data) < slow_period:
        return np.nan, np.nan, np.nan

    ema_fast = np.mean(data[-fast_period:])
    ema_slow = np.mean(data[-slow_period:])
    macd = ema_fast - ema_slow
    signal = calculate_sma([macd], signal_period) # Using SMA for signal line calculation
    histogram = macd - signal if not np.isnan(signal) else np.nan
    return macd, signal, histogram
