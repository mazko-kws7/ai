df = pd.DataFrame(daily_data, columns=['Open', 'High', 'Low', 'Close', 'Volume'])

# Calculate 20-day Moving Average
df['MA_20'] = df['Close'].rolling(window=20).mean()

# Calculate 14-day RSI
def calculate_rsi(data, window=14):
    delta = data['Close'].diff()
    up = delta.clip(lower=0)
    down = -1 * delta.clip(upper=0)
    
    avg_up = up.rolling(window=window, min_periods=window).mean()
    avg_down = down.rolling(window=window, min_periods=window).mean()
    
    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    return rsi

df['RSI'] = calculate_rsi(df)

# Calculate MACD (12-day EMA, 26-day EMA, 9-day EMA of difference)
df['EMA_12'] = df['Close'].ewm(span=12, adjust=False).mean()
df['EMA_26'] = df['Close'].ewm(span=26, adjust=False).mean()
df['MACD'] = df['EMA_12'] - df['EMA_26']
df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()

print(df.tail(5))
