def analyze_daily_data(data):
    close_prices = [float(x.split(',')[4]) for x in data.strip().split('\n')]
    
    # Calculate 7-day SMA
    sma_7 = []
    for i in range(6, len(close_prices)):
        sma = sum(close_prices[i-6:i+1]) / 7
        sma_7.append(sma)
    
    # Approximate 5-day RSI
    rsi_5 = []
    for i in range(5, len(close_prices)):
        gains = 0
        losses = 0
        for j in range(1, 6):
            change = close_prices[i-j+1] - close_prices[i-j]
            if change > 0:
                gains += change
            else:
                losses -= change
        if gains + losses == 0:
            rs = 50  # Assign neutral value if no gains or losses
        else:
            rs = gains / losses
        rsi = 100 - (100 / (1 + rs))
        rsi_5.append(rsi)
    
    # Generate Signals (starting after SMA and RSI can be calculated)
    signals = []
    for i in range(max(6,5), len(close_prices)):
        current_price = close_prices[i]
        
        # Get the index of SMA and RSI based on i:
        sma_index = i - 6 if i >= 6 else None
        rsi_index = i - 5 if i >= 5 else None
        
        if sma_index is not None and rsi_index is not None:
            #Using the last calculated SMA and RSI values for signal generation
            previous_price = close_prices[i-1]
            current_sma = sma_7[-1]
            current_rsi = rsi_5[-1]

            if previous_price < current_sma and current_price > current_sma and current_rsi < 30:
                signals.append({
                    "Signal Type": "Buy",
                    "Cryptocurrency": "LEO-USD",
                    "Confidence Level": "Medium",
                    "Take Profit Probability": "60%",
                    "Justification": "Price crossed above 7-day SMA, RSI indicates potential oversold condition.",
                    "Relevant Data": f"Price: {current_price}, 7-day SMA: {current_sma}, RSI: {current_rsi}",
                    "Recommended Action": "Consider buying a small position."
                })
            elif previous_price > current_sma and current_price < current_sma and current_rsi > 70:
                signals.append({
                    "Signal Type": "Sell",
                    "Cryptocurrency": "LEO-USD",
                    "Confidence Level": "Medium",
                    "Take Profit Probability": "65%",
                    "Justification": "Price crossed below 7-day SMA, RSI indicates potential overbought condition.",
                    "Relevant Data": f"Price: {current_price}, 7-day SMA: {current_sma}, RSI: {current_rsi}",
                    "Recommended Action": "Consider selling a portion of holdings."
                })
            
    return signals
