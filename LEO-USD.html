def calculate_sma(data, period):
    if len(data) < period:
        return None
    return sum(data[-period:]) / period

def calculate_rsi(data, period):
    if len(data) < period:
        return None
    
    deltas = [data[i] - data[i-1] for i in range(1, len(data))]
    
    if len(deltas) < period:
        return None

    avg_gain = sum([d for d in deltas[-period:] if d > 0]) / period
    avg_loss = abs(sum([d for d in deltas[-period:] if d < 0]) / period)

    if avg_loss == 0:
        return 100
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def calculate_macd(data, fast_period=12, slow_period=26):
    if len(data) < slow_period:
        return None, None
    
    ema_fast = calculate_ema(data, fast_period)
    ema_slow = calculate_ema(data, slow_period)
    
    if ema_fast is None or ema_slow is None:
        return None, None

    macd = ema_fast - ema_slow
    
    # Calculate signal line (9-day EMA of MACD)
    signal_line_data = [macd]  # Using the last calculated MACD value
    signal_line = calculate_ema(signal_line_data, 9)  # 9-day EMA
    
    return macd, signal_line

def calculate_ema(data, period):
    if len(data) < period:
        return None
    
    ema = sum(data[:period]) / period
    
    for i in range(period, len(data)):
        ema = (data[i] * (2 / (period + 1))) + ema * (1 - (2 / (period + 1)))
        
    return ema
