import numpy as np
import pandas as pd

# Simple Moving Average
def SMA(data, period):
    return pd.Series(data).rolling(window=period).mean()

# Exponential Moving Average
def EMA(data, period):
    return pd.Series(data).ewm(span=period, adjust=False).mean()

# Relative Strength Index
def RSI(data, period):
    delta = data.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    avg_gain = up.rolling(window=period).mean()
    avg_loss = abs(down.rolling(window=period).mean())

    RS = avg_gain / avg_loss
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

# MACD
def MACD(data, fast_period, slow_period, signal_period):
    EMA_fast = EMA(data, fast_period)
    EMA_slow = EMA(data, slow_period)
    MACD = EMA_fast - EMA_slow
    signal = EMA(MACD, signal_period)
    return MACD, signal
