import numpy as np

def calculate_rsi(prices, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = np.diff(prices)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    roll_up1 = np.mean(up[:period])
    roll_down1 = np.mean(np.abs(down[:period]))
    
    up_list = [roll_up1]
    down_list = [roll_down1]
    
    for i in range(period, len(up)):
        roll_up = (up_list[-1] * (period - 1) + up[i]) / period
        roll_down = (down_list[-1] * (period - 1) + np.abs(down[i])) / period
        up_list.append(roll_up)
        down_list.append(roll_down)
    
    rs = np.array(up_list) / np.array(down_list)
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_sma(prices, period=20):
    """Calculates the Simple Moving Average (SMA)."""
    if len(prices) < period:
        return np.nan
    return np.mean(prices[-period:])

def calculate_ema(prices, period=20):
    """Calculates the Exponential Moving Average (EMA)."""
    if len(prices) < period:
        return np.nan
    
    ema = np.zeros_like(prices, dtype=float)
    ema[:period] = np.mean(prices[:period])
    k = 2 / (period + 1)
    for i in range(period, len(prices)):
        ema[i] = prices[i] * k + ema[i-1] * (1 - k)
    return ema[-1]
