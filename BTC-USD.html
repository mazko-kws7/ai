def calculate_sma(data, period): sma = pd.Series(data[Close].rolling(window=period).mean(), name=SMA_ + str(period)) return sma

def calculate_rsi(data, period=14): delta = data[Close].diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 down = abs(down) roll_up = up.rolling(window=period).mean() roll_down = down.rolling(window=period).mean() rs = roll_up / roll_down rsi = 100.0 - (100.0 / (1.0 + rs)) return pd.Series(rsi, name=RSI_ + str(period))

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9): ema_fast = data[Close].ewm(span=fast_period, adjust=False).mean() ema_slow = data[Close].ewm(span=slow_period, adjust=False).mean() macd = ema_fast - ema_slow signal = macd.ewm(span=signal_period, adjust=False).mean() histogram = macd - signal return macd, signal, histogram
