def calculate_ma(data, period):
    """Calculates the moving average of a list of closing prices."""
    closing_prices = [float(row.split(',')[4]) for row in data.strip().split('\n')]
    if len(closing_prices) < period:
        return None
    ma = sum(closing_prices[-period:]) / period
    return ma
