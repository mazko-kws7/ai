def calculate_rsi(data, period=14): ""Calculates the Relative Strength Index (RSI)."" delta = data[Close].diff() delta = delta[1:] # Remove the first NaN value

up, down = delta.copy(), delta.copy()
up[up < 0] = 0
down[down > 0] = 0
down = -down

avg_up = up.rolling(window=period, min_periods=period).mean()
avg_down = down.rolling(window=period, min_periods=period).mean()

rs = avg_up / avg_down
rsi = 100 - (100 / (1 + rs))
return rsi
