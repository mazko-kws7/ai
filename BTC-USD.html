def calculate_sma(data, period):
    sma = []
    for i in range(len(data)):
        if i < period - 1:
            sma.append(None)  # Not enough data for SMA
        else:
            window = data[i - period + 1:i + 1]
            sma.append(sum(window) / period)
    return sma

def calculate_rsi(data, period):
    gains = []
    losses = []
    for i in range(1, len(data)):
        change = data[i] - data[i - 1]
        if change > 0:
            gains.append(change)
            losses.append(0)
        else:
            gains.append(0)
            losses.append(abs(change))
    
    avg_gain = []
    avg_loss = []
    
    for i in range(len(gains)):
        if i < period - 1:
            avg_gain.append(None)
            avg_loss.append(None)
        else:
            avg_gain_window = gains[i - period + 1:i + 1]
            avg_loss_window = losses[i - period + 1:i + 1]
            avg_gain.append(sum(avg_gain_window) / period)
            avg_loss.append(sum(avg_loss_window) / period)

    rsi = []
    for i in range(len(avg_gain)):
        if avg_gain[i] is None or avg_loss[i] is None:
            rsi.append(None)
        elif avg_loss[i] == 0:
            rsi.append(100)
        else:
            rs = avg_gain[i] / avg_loss[i]
            rsi_val = 100 - (100 / (1 + rs))
            rsi.append(rsi_val)

    return [None]*period + rsi #shift the rsi values to align correctly with the original data length
