# 7-day SMA calculation (example)
def calculate_sma(prices, period):
    if len(prices) < period:
        return None
    return sum(prices[-period:]) / period

# 14-day RSI Calculation (example - requires more historical data)
def calculate_rsi(prices, period=14):
    if len(prices) <= period:
        return None

    gains = []
    losses = []

    for i in range(1, len(prices)):
        change = prices[i] - prices[i - 1]
        if change > 0:
            gains.append(change)
            losses.append(0)
        else:
            losses.append(abs(change))
            gains.append(0)

    avg_gain = sum(gains[-(period):]) / period if len(gains) >= period else 0
    avg_loss = sum(losses[-(period):]) / period if len(losses) >= period else 0

    if avg_loss == 0:
        return 100 if avg_gain > 0 else 0  # Avoid division by zero
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi
