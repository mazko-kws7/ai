def calculate_rsi(prices, period=14): if len(prices) <= period: return np.nan

deltas = np.diff(prices)
seed = deltas[:period+1]
up = seed[seed >= 0].sum()/period
down = -seed[seed < 0].sum()/period
rs = up/down
return 100 - (100/(1+rs))
