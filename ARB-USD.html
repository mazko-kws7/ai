# 14-day RSI
def calculate_rsi(data, period=14):
    delta = np.diff(data)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = calculate_sma(up, period)[period-1:]
    avg_loss = np.abs(calculate_sma(down, period))[period-1:]
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi[-1]

rsi_14 = calculate_rsi(close_prices)
print("14-day RSI:", rsi_14)
