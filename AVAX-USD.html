import pandas as pd
import numpy as np

def calculate_sma(data, period):
    """
    Calculates the Simple Moving Average (SMA) for a given dataset.

    Args:
        data (pd.Series): Time series data for which SMA is to be calculated.
        period (int): The period over which SMA is calculated.

    Returns:
        pd.Series: SMA values for the given data.
    """
    return data.rolling(window=period).mean()

def calculate_rsi(close, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given series of closing prices.

    Args:
        close (pd.Series): A pandas Series of closing prices.
        period (int): The period over which RSI is calculated (default: 14).

    Returns:
        pd.Series: A pandas Series of RSI values.
    """
    delta = close.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = up.rolling(window=period, min_periods=period).mean()
    avg_loss = abs(down.rolling(window=period, min_periods=period).mean())

    rs = avg_gain / avg_loss
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_macd(close, fast_period=12, slow_period=26, signal_period=9):
    """
    Calculates the Moving Average Convergence Divergence (MACD) for a given series of closing prices.

    Args:
        close (pd.Series): A pandas Series of closing prices.
        fast_period (int): The period for the fast EMA (default: 12).
        slow_period (int): The period for the slow EMA (default: 26).
        signal_period (int): The period for the signal line EMA (default: 9).

    Returns:
        tuple: A tuple containing three pandas Series: MACD line, signal line, and MACD histogram.
    """
    ema_fast = close.ewm(span=fast_period, adjust=False).mean()
    ema_slow = close.ewm(span=slow_period, adjust=False).mean()
    macd_line = ema_fast - ema_slow
    signal_line = macd_line.ewm(span=signal_period, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    return macd_line, signal_line, macd_histogram
