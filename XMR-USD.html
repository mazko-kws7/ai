def calculate_rsi(data, period=14): ""“Calculates the Relative Strength Index (RSI).”"" delta = np.diff(data) up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 down = -down roll_up = np.mean(up[:period]) roll_down = np.mean(down[:period])

rs = roll_up / roll_down
rsi = 100.0 - (100.0 / (1.0 + rs))
return rsi

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9): ""“Calculates the Moving Average Convergence Divergence (MACD).”"" ema_fast = np.mean(data[:fast_period]) ema_slow = np.mean(data[:slow_period]) macd = ema_fast - ema_slow signal = np.mean(np.array([macd]*signal_period)) return macd, signal

def analyze_daily_chart(daily_data): ""“Analyzes the daily chart data.”"" closes = np.array([float(x.split(‘,’)[4]) for x in daily_data.strip().split(‘’)]) rsi = calculate_rsi(closes) macd, signal = calculate_macd(closes) # Trend Analysis (Simple Moving Average) sma_short = np.mean(closes[-20:]) # 20-day SMA sma_long = np.mean(closes)

return rsi, macd, signal, sma_short, sma_long

def analyze_hourly_chart(hourly_data): ""“Analyzes the hourly chart data.”"" closes = np.array([float(x.split(‘,’)[4]) for x in hourly_data.strip().split(‘’)]) rsi = calculate_rsi(closes) macd, signal = calculate_macd(closes)

return rsi, macd, signal, sma_short, sma_long
