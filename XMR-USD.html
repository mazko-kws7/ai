def calculate_rsi(prices, period=14):
    deltas = []
    for i in range(1, len(prices)):
        deltas.append(prices[i] - prices[i-1])
    
    avg_gain = 0
    avg_loss = 0
    
    # Initial average gain and loss
    for i in range(period):
        if deltas[i] > 0:
            avg_gain += deltas[i]
        else:
            avg_loss += abs(deltas[i])
    avg_gain /= period
    avg_loss /= period
    
    rsi_values = []
    
    # Calculate RSI for the rest of the series
    for i in range(period, len(prices)-1):
        gain = deltas[i] if deltas[i] > 0 else 0
        loss = abs(deltas[i]) if deltas[i] < 0 else 0
        
        avg_gain = (avg_gain * (period - 1) + gain) / period
        avg_loss = (avg_loss * (period - 1) + loss) / period
        
        if avg_loss == 0:
            rsi = 100  # Avoid division by zero if no losses
        else:
            rs = avg_gain / avg_loss
            rsi = 100 - (100 / (1 + rs))
            
        rsi_values.append(rsi)
        
    return rsi_values
def calculate_sma(prices, period):
    if len(prices) < period:
        return None
    sma_values = []
    for i in range(period, len(prices) + 1):
        sma = sum(prices[i-period:i]) / period
        sma_values.append(sma)
    return sma_values
