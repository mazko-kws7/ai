def calculate_ma(data, period):
    ma = []
    for i in range(len(data)):
        if i < period - 1:
            ma.append(None)  # Not enough data to calculate MA
        else:
            window = data[i - period + 1:i + 1]
            average = sum(window) / period
            ma.append(average)
    return ma
