# Calculate RSI
def calculate_rsi(data, period=14):
    delta = np.diff(data)
    up, down = delta.copy(), delta.copy()
    up[up<0] = 0
    down[down>0] = 0
    
    rol_up = np.mean(up[-period:])
    rol_down = np.mean(np.abs(down[-period:]))
    
    rs = rol_up / rol_down
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi
