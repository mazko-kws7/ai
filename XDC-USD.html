import numpy as np

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    if len(data) < period:
        return np.nan
    return np.mean(data[-period:])

def calculate_rsi(prices, period=14):
    """Calculates Relative Strength Index."""
    if len(prices) < period + 1:
        return np.nan

    deltas = np.diff(prices)
    seed = deltas[:period+1]
    up = seed[seed >= 0].sum()/period
    down = -seed[seed < 0].sum()/period
    rs = up/down
    rsi = np.where(down == 0, 100, 100 - (100 / (1 + rs)))
    return rsi[-1]

def assess_trend(prices):
    """Simple trend assessment: Upward, Downward, or Sideways."""
    if len(prices) < 3:
        return "Unclear"

    # Compare the most recent price to the price from two periods ago
    if prices[-1] > prices[-3]:
        return "Upward"
    elif prices[-1] < prices[-3]:
        return "Downward"
    else:
        return "Sideways"
