daily_df = pd.DataFrame(daily_data, columns=['Open', 'High', 'Low', 'Close', 'Volume'])

# Calculate 20-day SMA
daily_df['SMA_20'] = daily_df['Close'].rolling(window=20).mean()

# Calculate RSI (Relative Strength Index)
def calculate_rsi(data, window=14):
    delta = data['Close'].diff()
    up = delta.clip(lower=0)
    down = -1 * delta.clip(upper=0)
    
    avg_up = up.rolling(window=window, min_periods=window).mean()
    avg_down = down.rolling(window=window, min_periods=window).mean()

    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    return rsi

daily_df['RSI'] = calculate_rsi(daily_df)

# Calculate MACD (Moving Average Convergence Divergence) - Approximation
# Using simple moving averages instead of exponential for demonstration with given data
daily_df['EMA_12'] = daily_df['Close'].rolling(window=12).mean()
daily_df['EMA_26'] = daily_df['Close'].rolling(window=26).mean()
daily_df['MACD'] = daily_df['EMA_12'] - daily_df['EMA_26']
daily_df['Signal'] = daily_df['MACD'].rolling(window=9).mean()

print(daily_df.tail())
