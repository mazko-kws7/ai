def calculate_sma(df, period=50): df[SMA] = df[Close].rolling(window=period).mean() return df

def calculate_rsi(df, period=14): delta = df[Close].diff() up = delta.clip(lower=0) down = -1 * delta.clip(upper=0)

roll_up1 = up.ewm(alpha=1/period).mean()
roll_down1 = down.ewm(alpha=1/period).mean()

RS = roll_up1 / roll_down1
RSI = 100.0 - (100.0 / (1.0 + RS))
df['RSI'] = RSI
return df
