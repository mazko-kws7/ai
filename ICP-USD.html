import numpy as np
import pandas as pd

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    return pd.Series(data).rolling(window=period).mean().values

def calculate_ema(data, period):
    """Calculates Exponential Moving Average."""
    ema = [sum(data[:period]) / period]
    for i in range(period, len(data)):
        ema.append((data[i] * (2 / (period + 1))) + ema[-1] * (1 - (2 / (period + 1))))
    return np.array(ema)

def calculate_rsi(close, period=14):
    """Calculates Relative Strength Index."""
    delta = np.diff(close)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    avg_up = calculate_sma(up, period)
    avg_down = np.abs(calculate_sma(down, period))
    rs = avg_up / avg_down
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_macd(close, fast_period=12, slow_period=26, signal_period=9):
    """Calculates Moving Average Convergence Divergence."""
    ema_fast = calculate_ema(close, fast_period)
    ema_slow = calculate_ema(close, slow_period)
    macd_line = ema_fast[slow_period-1:] - ema_slow[slow_period-1:]
    signal_line = calculate_ema(macd_line, signal_period)
    return macd_line[signal_period-1:], signal_line[signal_period-1:]

def calculate_bollinger_bands(close, period=20, num_std=2):
  """Calculates Bollinger Bands."""
  sma = calculate_sma(close, period)
  std = pd.Series(close).rolling(window=period).std().values
  upper_band = sma + (std * num_std)
  lower_band = sma - (std * num_std)
  return upper_band, sma, lower_band

def calculate_fibonacci_retracement(high, low):
    """Calculates Fibonacci Retracement levels."""
    diff = high - low
    levels = [0.236, 0.382, 0.5, 0.618, 0.786]
    retracement_levels = [high - level * diff for level in levels]
    return retracement_levels
