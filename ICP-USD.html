# Trading Signal Generation Logic

def generate_signal(df, is_daily, scenario="Base Case"):
    """
    Generates a trading signal based on technical indicators and market scenario.
    """
    last_row = df.iloc[-1]
    current_price = last_row['Close']
    signal_type = None
    confidence_level = "Medium"  # Default
    take_profit_probability = 50  # Default
    justification = ""
    relevant_data = {}
    recommended_action = ""

    #Determine timeframe for reporting
    timeframe = "Daily" if is_daily else "Hourly"

    #Buy Signal Condition
    if last_row['RSI_14'] < 30 and last_row['MACD'] > last_row['Signal']:
        signal_type = "Buy"
        justification = f"Oversold RSI ({last_row['RSI_14']:.2f}) and MACD crossover indicate potential upward movement. "
        relevant_data = {"RSI": f"{last_row['RSI_14']:.2f}", "MACD": f"{last_row['MACD']:.3f}", "Signal": f"{last_row['Signal']:.3f}"}
        recommended_action = "Consider buying a small position."

    #Sell Signal Condition
    elif last_row['RSI_14'] > 70 and last_row['MACD'] < last_row['Signal']:
        signal_type = "Sell"
        justification = f"Overbought RSI ({last_row['RSI_14']:.2f}) and MACD bearish crossover suggest potential downward correction. "
        relevant_data = {"RSI": f"{last_row['RSI_14']:.2f}", "MACD": f"{last_row['MACD']:.3f}", "Signal": f"{last_row['Signal']:.3f}"}
        recommended_action = "Consider selling a portion of holdings."
    
    #Additional logic based on SMA
    elif current_price > last_row['SMA_20']:
        signal_type = "Buy"
        justification = f"Price is above 20-period SMA, suggesting bullish momentum. "
        relevant_data = {"Price": f"{current_price:.2f}", "SMA_20": f"{last_row['SMA_20']:.2f}"}
        recommended_action = "Consider a long position."
    else:
        signal_type = "Sell"
        justification = f"Price is below 20-period SMA, indicating bearish momentum. "
        relevant_data = {"Price": f"{current_price:.2f}", "SMA_20": f"{last_row['SMA_20']:.2f}"}
        recommended_action = "Consider exiting long positions or shorting."

    # Scenario Adjustments
    if scenario == "Positive News":
        confidence_level = "High"
        take_profit_probability = 75
        justification += "Positive regulatory news strengthens the bullish outlook."
    elif scenario == "Negative News":
        confidence_level = "Low"
        take_profit_probability = 25
        justification += "Security breach dampens market sentiment; trade with caution."

    return {
        "Signal Type": signal_type,
        "Cryptocurrency": "ICP-USD",
        "TimeFrame": timeframe,
        "Confidence Level": confidence_level,
        "Take Profit Probability": take_profit_probability,
        "Justification": justification,
        "Relevant Data": relevant_data,
        "Recommended Action": recommended_action
    }
