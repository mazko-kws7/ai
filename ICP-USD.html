# Calculate RSI (14-day)
def calculate_rsi(data, window=14):
    delta = data['Close'].diff()
    up = delta.clip(lower=0)
    down = -1 * delta.clip(upper=0)
    
    avg_up = up.rolling(window=window, min_periods=window).mean()
    avg_down = down.rolling(window=window, min_periods=window).mean()

    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    return rsi

df_daily['RSI'] = calculate_rsi(df_daily)

# Calculate MACD (12, 26, 9)
def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    ema_fast = data['Close'].ewm(span=fast_period, adjust=False).mean()
    ema_slow = data['Close'].ewm(span=slow_period, adjust=False).mean()
    macd = ema_fast - ema_slow
    signal = macd.ewm(span=signal_period, adjust=False).mean()
    histogram = macd - signal
    return macd, signal, histogram

macd, signal, histogram = calculate_macd(df_daily)
df_daily['MACD'] = macd
df_daily['Signal'] = signal
df_daily['Histogram'] = histogram
