def calculate_sma(df, period=20): return df[‘Close’].rolling(window=period).mean()

def calculate_rsi(df, period=14): delta = df[‘Close’].diff() up = delta.clip(lower=0) down = -1 * delta.clip(upper=0)

avg_up = up.rolling(window=period).mean()
avg_down = down.rolling(window=period).mean()

rs = avg_up / avg_down
rsi = 100 - (100 / (1 + rs))
return rsi
