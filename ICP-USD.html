def calculate_rsi(data, period=14): ""“Calculates the Relative Strength Index (RSI).”"" if len(data) < period + 1: return np.nan

deltas = np.diff(data)
seed = deltas[:period+1]
up = seed[seed >= 0].sum()/period
down = -seed[seed < 0].sum()/period
rs = up/down
rsi = np.zeros_like(data)
rsi[:period] = np.nan
rsi[period] = 100. - 100./(1.+rs)

for i in range(period+1, len(data)):
    delta = deltas[i-1] # the diff is one index shorter

    if delta > 0:
        upval = delta
        downval = 0.
    else:
        upval = 0.
        downval = -delta

    up = (up*(period-1) + upval)/period
    down = (down*(period-1) + downval)/period

    rs = up/down
    rsi[i] = 100. - 100./(1.+rs)

return rsi
