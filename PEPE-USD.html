import numpy as np
import pandas as pd

def calculate_moving_average(data, window):
    """Calculates the moving average of a given dataset.

    Args:
        data (list): A list of numerical data.
        window (int): The window size for the moving average.

    Returns:
        list: A list of moving average values.
    """
    if len(data) < window:
        return [np.nan] * len(data)  # Not enough data to calculate MA
    return pd.Series(data).rolling(window=window).mean().tolist()

def calculate_rsi(data, window):
    """Calculates the Relative Strength Index (RSI) of a given dataset.

    Args:
        data (list): A list of numerical data (e.g., closing prices).
        window (int): The window size for the RSI calculation.

    Returns:
        list: A list of RSI values.
    """
    if len(data) < window:
        return [np.nan] * len(data)

    delta = pd.Series(data).diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.rolling(window=window).mean()
    roll_down1 = np.abs(down.rolling(window=window).mean())
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI.tolist()

def analyze_trend(data):
  """Analyzes the trend of the given data.

  Args:
    data: A list of numerical data.

  Returns:
    A string indicating the trend: "Upward", "Downward", or "Sideways".
  """
  if len(data) < 3:
    return "Insufficient Data"

  # Calculate a simple linear regression
  x = np.arange(len(data))
  slope, intercept = np.polyfit(x, data, 1)

  if slope > 0.0000001:  # Add a threshold to avoid noise
    return "Upward"
  elif slope < -0.0000001: # Add a threshold to avoid noise
    return "Downward"
  else:
    return "Sideways"
