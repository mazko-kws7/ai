import numpy as np

def calculate_moving_average(data, window):
    """Calculates the moving average of a given dataset."""
    if len(data) < window:
        return np.array([])
    return np.convolve(data, np.ones(window), 'valid') / window

def calculate_rsi(data, window):
    """Calculates the Relative Strength Index (RSI)."""
    if len(data) < window:
        return np.array([])
    deltas = np.diff(data)
    seed = deltas[:window+1]
    up = seed[seed >= 0].sum() / window
    down = -seed[seed < 0].sum() / window
    rs = up / down
    rsi = np.zeros_like(data)
    rsi[:window] = 100. - 100. / (1. + rs)

    for i in range(window, len(data)):
        delta = deltas[i - 1]  # Changed index to i-1
        if delta > 0:
            upval = delta
            downval = 0.
        else:
            upval = 0.
            downval = -delta

        up = (up * (window - 1) + upval) / window
        down = (down * (window - 1) + downval) / window

        rs = up / down
        rsi[i] = 100. - 100. / (1. + rs)

    return rsi[window:]  # Adjusted return to match valid indices


def identify_support_resistance(data, window):
    """Identifies support and resistance levels based on local minima and maxima."""
    support_levels = []
    resistance_levels = []
    for i in range(1, len(data) - 1):
        if data[i-1] > data[i] < data[i+1]:
            support_levels.append((i, data[i]))
        elif data[i-1] < data[i] > data[i+1]:
            resistance_levels.append((i, data[i]))
    return support_levels, resistance_levels

def analyze_volume(volume_data, price_data):
    """Analyzes volume in relation to price movements."""
    volume_trends = []
    for i in range(1, len(volume_data)):
        if price_data[i] > price_data[i-1] and volume_data[i] > volume_data[i-1]:
            volume_trends.append("Bullish Volume")
        elif price_data[i] < price_data[i-1] and volume_data[i] > volume_data[i-1]:
            volume_trends.append("Bearish Volume")
        else:
            volume_trends.append("Neutral Volume")
    return volume_trends
