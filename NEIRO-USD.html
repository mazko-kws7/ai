def calculate_ema(data, period=12): ""“Calculates the Exponential Moving Average (EMA)”"" ema = np.zeros_like(data, dtype=float) ema[:period] = np.mean(data[:period]) smoothing_factor = 2 / (period + 1)

for i in range(period, len(data)): ema[i] = (data[i] * smoothing_factor) + (ema[i-1] * (1 - smoothing_factor)) return ema[period-1:]
