import numpy as np

def calculate_moving_average(data, window):
  """Calculates the moving average of a data series.

  Args:
    data: A list of numerical data.
    window: The window size for the moving average.

  Returns:
    A list of moving average values.
  """
  if len(data) < window:
    return [np.nan] * len(data)
  
  moving_average = np.convolve(data, np.ones(window), 'valid') / window
  # Pad the beginning of the moving_average list with NaN values to match the length of the original data
  padding = [np.nan] * (window - 1)
  moving_average = padding + list(moving_average)
  return moving_average


def calculate_rsi(data, window):
  """Calculates the Relative Strength Index (RSI).

  Args:
    data: A list of numerical data (e.g., closing prices).
    window: The window size for the RSI calculation.

  Returns:
    A list of RSI values.
  """
  if len(data) < window:
        return [np.nan] * len(data)

  deltas = np.diff(data)
  seed = deltas[:window]
  up = seed[seed >= 0].sum() / window
  down = -seed[seed < 0].sum() / window
  rs = up / down
  rsi = np.zeros_like(data)
  rsi[:window] = 100. - 100. / (1. + rs)

  for i in range(window, len(data)):
    delta = deltas[i - 1]  # The correct index for deltas
    if delta > 0:
      upval = delta
      downval = 0.
    else:
      upval = 0.
      downval = -delta

    up = (up * (window - 1) + upval) / window
    down = (down * (window - 1) + downval) / window

    rs = up / down
    rsi[i] = 100. - 100. / (1. + rs)
  rsi = [np.nan] * (window - 1) + list(rsi[window-1:])
  return rsi


def analyze_trend(data):
    """Analyzes the trend of a data series.

    Args:
        data: A list of numerical data.

    Returns:
        A string indicating the trend ("Uptrend", "Downtrend", or "Sideways").
    """
    if len(data) < 3:
        return "Insufficient data"

    # Calculate a simple moving average for trend detection
    sma = calculate_moving_average(data, window=3)
    
    if np.isnan(sma[-1]) or np.isnan(sma[-2]):
        return "Insufficient data"

    if sma[-1] > sma[-2]:
        return "Uptrend"
    elif sma[-1] < sma[-2]:
        return "Downtrend"
    else:
        return "Sideways"
