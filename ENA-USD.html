def calculate_rsi(closes, period=14): if len(closes) <= period: return [None] * len(closes)

deltas = [closes[i] - closes[i-1] for i in range(1, len(closes))]

# Initialize gains and losses lists
avg_gain = [0] * len(closes)
avg_loss = [0] * len(closes)

# Calculate initial average gain and loss
up, down = [], []
for d in deltas[:period]:
    if d > 0:
        up.append(d)
        down.append(0)
    else:
        up.append(0)
        down.append(abs(d))
avg_gain[period] = sum(up) / period
avg_loss[period] = sum(down) / period

# Calculate subsequent average gains and losses
for i in range(period+1, len(closes)):
    delta = closes[i] - closes[i-1]
    gain = delta if delta > 0 else 0
    loss = abs(delta) if delta < 0 else 0
    avg_gain[i] = (avg_gain[i-1] * (period - 1) + gain) / period
    avg_loss[i] = (avg_loss[i-1] * (period - 1) + loss) / period

# Calculate RSI values
rsi = [0] * len(closes)
for i in range(period+1, len(closes)):
    if avg_loss[i] == 0:
        rsi[i] = 100  # Avoid division by zero
    else:
        rs = avg_gain[i] / avg_loss[i]
        rsi[i] = 100 - (100 / (1 + rs))

# Pad the RSI list with None values for the initial periods
rsi[:period+1] = [None] * (period+1)

return rsi
