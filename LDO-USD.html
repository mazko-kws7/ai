def calculate_rsi(data, period=14): ""“Calculates the Relative Strength Index (RSI).”"" delta = np.diff(data) up, down = delta.copy(), delta.copy() up[up<0] = 0 down[down>0] = 0

roll_up1 = np.mean(up[:period])
roll_down1 = np.mean(np.abs(down[:period]))

up_list = [roll_up1]
down_list = [roll_down1]

for i in range(period, len(up)):
    up_change = up[i]
    down_change = np.abs(down[i])
    
    roll_up = (roll_up1 * (period - 1) + up_change) / period
    roll_down = (roll_down1 * (period - 1) + down_change) / period
    
    up_list.append(roll_up)
    down_list.append(roll_down)
    
    roll_up1 = roll_up
    roll_down1 = roll_down

rs = np.array(up_list) / np.array(down_list)
rsi = 100.0 - (100.0 / (1.0 + rs))
return rsi
