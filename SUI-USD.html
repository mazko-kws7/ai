def analyze_data(data):
    """Calculates simple moving averages and RSI.

    Args:
        data: A list of lists, where each inner list represents a day's OHLCV data.

    Returns:
        A tuple containing the 14-day SMA, 50-day SMA, and 14-day RSI.
    """
    closes = [row[4] for row in data]
    
    # Simple Moving Average (SMA) calculation
    def calculate_sma(period):
        if len(closes) < period:
            return None
        sma_values = []
        for i in range(period, len(closes) + 1):
            sma_values.append(sum(closes[i-period:i]) / period)
        return sma_values[-1] if sma_values else None

    sma_14 = calculate_sma(14)
    sma_50 = calculate_sma(50)

    # Relative Strength Index (RSI) calculation
    def calculate_rsi(period):
        if len(closes) < period + 1:
            return None
        
        price_changes = [closes[i] - closes[i-1] for i in range(1, len(closes))]
        gains = [change for change in price_changes if change > 0]
        losses = [-change for change in price_changes if change < 0]  # Use absolute value for losses

        if not gains:
            avg_gain = 0
        else:
            avg_gain = sum(gains[-period:]) / min(len(gains), period)
        
        if not losses:
            avg_loss = 0
        else:
            avg_loss = sum(losses[-period:]) /  min(len(losses), period)

        if avg_loss == 0:
            return 100  # Special case: all gains
        
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    rsi_14 = calculate_rsi(14)

    return sma_14, sma_50, rsi_14
