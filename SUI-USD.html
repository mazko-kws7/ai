def calculate_rsi(data, period):
    if len(data) <= period:
        return None
    
    diffs = [data[i] - data[i-1] for i in range(1, len(data))]
    
    # Calculate average gains and losses over the period
    gains = [d for d in diffs[-period:] if d > 0]
    losses = [-d for d in diffs[-period:] if d < 0]
    
    avg_gain = sum(gains) / period if gains else 0
    avg_loss = sum(losses) / period if losses else 0
    
    if avg_loss == 0:
        return 100 if avg_gain > 0 else 0  # Avoid division by zero
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def analyze_macd(data, fast_period=12, slow_period=26, signal_period=9):
    if len(data) <= slow_period:
        return None, None

    # Calculate EMAs (Exponential Moving Averages)
    ema_fast = calculate_ema(data, fast_period)
    ema_slow = calculate_ema(data, slow_period)

    # Calculate MACD line
    macd_line = ema_fast[-1] - ema_slow[-1] if ema_fast and ema_slow else None

    # Calculate Signal line (9-day EMA of MACD line)
    if macd_line is not None:
        signal_line = calculate_ema([macd_line], signal_period)[-1] if calculate_ema([macd_line], signal_period) else None
    else:
        signal_line = None

    return macd_line, signal_line

def calculate_ema(data, period):
    if len(data) < period:
        return None

    ema = [sum(data[:period]) / period]  # Simple Moving Average as initial value

    # Calculate smoothing factor
    smoothing_factor = 2 / (period + 1)

    # Calculate EMA for the rest of the data
    for i in range(period, len(data)):
        ema.append((data[i] * smoothing_factor) + (ema[-1] * (1 - smoothing_factor)))

    return ema
