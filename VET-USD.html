# Calculate RSI (14-day)
def calculate_rsi(data, period=14):
    delta = data['Close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    rol_up = up.rolling(period).mean()
    rol_down = down.abs().rolling(period).mean()
    
    RS = rol_up / rol_down
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

df['RSI'] = calculate_rsi(df)

# Calculate MACD (12, 26, 9)
def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    EMA_fast = data['Close'].ewm(span=fast_period, adjust=False).mean()
    EMA_slow = data['Close'].ewm(span=slow_period, adjust=False).mean()
    MACD = EMA_fast - EMA_slow
    Signal = MACD.ewm(span=signal_period, adjust=False).mean()
    Histogram = MACD - Signal
    return MACD, Signal, Histogram

df['MACD'], df['Signal'], df['Histogram'] = calculate_macd(df)

# Calculate Bollinger Bands (20-day)
def calculate_bollinger_bands(data, period=20, num_std=2):
    rolling_mean = data['Close'].rolling(period).mean()
    rolling_std = data['Close'].rolling(period).std()
    upper_band = rolling_mean + (rolling_std * num_std)
    lower_band = rolling_mean - (rolling_std * num_std)
    return upper_band, rolling_mean, lower_band

df['Upper_Band'], df['Middle_Band'], df['Lower_Band'] = calculate_bollinger_bands(df)
