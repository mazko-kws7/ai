def calculate_moving_averages(data, short_window, long_window): """Calculates short and long term moving averages.

Args:
    data (list): List of closing prices.
    short_window (int): Period for the short-term moving average.
    long_window (int): Period for the long-term moving average.

Returns:
    tuple: Short-term moving average, Long-term moving average.
"""
short_ma = np.mean(data[-short_window:]) if len(data) >= short_window else None
long_ma = np.mean(data[-long_window:]) if len(data) >= long_window else None
return short_ma, long_ma

def calculate_rsi(data, window): """Calculates the Relative Strength Index (RSI).

Args:
    data (list): List of closing prices.
    window (int): Period for the RSI calculation.

Returns:
    float: RSI value.
"""
if len(data) < window:
    return None

deltas = np.diff(data)
seed = deltas[:window]
up = seed[seed >= 0].sum() / window
down = -seed[seed < 0].sum() / window
rs = up / down if down != 0 else 0
rsi = 100 - 100 / (1 + rs)

for i in range(window, len(deltas)):
    delta = deltas[i]
    if delta >= 0:
        up = (up * (window - 1) + delta) / window
        down = down * (window - 1) / window
    else:
        up = up * (window - 1) / window
        down = (down * (window - 1) - delta) / window

    rs = up / down if down != 0 else 0
    rsi = 100 - 100 / (1 + rs)

return rsi
