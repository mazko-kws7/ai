def calculate_sma(data, period):
    if len(data) < period:
        return None
    return np.mean(data[-period:])

def calculate_rsi(data, period=14):
    if len(data) < period + 1:
        return None
    
    deltas = np.diff(data)
    seed = deltas[:period+1]
    up = seed[seed >= 0].sum() / period
    down = -seed[seed < 0].sum() / period
    rs = up / down
    rsi = 100. - 100. / (1. + rs)

    for i in range(period + 1, len(data)):
        delta = deltas[i - 1]  # Corrected index
        if delta > 0:
            upval = delta
            downval = 0.
        else:
            upval = 0.
            downval = -delta

        up = (up * (period - 1) + upval) / period
        down = (down * (period - 1) + downval) / period
        rs = up / down
        rsi = 100. - 100. / (1. + rs)
    return rsi

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    if len(data) < slow_period:
        return None, None, None

    ema_fast = np.mean(data[-fast_period:])
    ema_slow = np.mean(data[-slow_period:])
    macd_line = ema_fast - ema_slow
    signal_line = np.mean([macd_line] * signal_period) # Use the same value for all periods in the signal line

    return macd_line, signal_line, macd_line - signal_line  # MACD, Signal, Histogram
