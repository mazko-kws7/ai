import numpy as np
import pandas as pd

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = data.diff()
    delta = delta[1:]
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.rolling(period).mean()
    roll_down1 = down.abs().rolling(period).mean()
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

def calculate_sma(data, period=20):
    """Calculates the Simple Moving Average (SMA)."""
    return data.rolling(window=period).mean()

def calculate_ema(data, period=12):
    """Calculates the Exponential Moving Average (EMA)."""
    return data.ewm(span=period, adjust=False).mean()

def assess_trend(data, period=5):
    """Assess the trend based on the last 'period' of data."""
    if data.iloc[-1] > data.iloc[-period]:
        return "Uptrend"
    elif data.iloc[-1] < data.iloc[-period]:
        return "Downtrend"
    else:
        return "Sideways"
