import numpy as np

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = np.diff(data)
    up, down = delta.copy(), delta.copy()
    up[up<0] = 0
    down[down>0] = 0

    roll_up1 = pd.Series(up).rolling(period).mean()
    roll_down1 = pd.Series(down).abs().rolling(period).mean()

    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI.values

def calculate_sma(data, period=20):
    """Calculates the Simple Moving Average (SMA)."""
    return pd.Series(data).rolling(window=period).mean().values

def calculate_ema(data, period=20):
    """Calculates the Exponential Moving Average (EMA)."""
    return pd.Series(data).ewm(span=period, adjust=False).mean().values

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    ema_fast = calculate_ema(data, fast_period)
    ema_slow = calculate_ema(data, slow_period)
    macd_line = ema_fast - ema_slow
    signal_line = calculate_ema(macd_line, signal_period)
    return macd_line, signal_line

def calculate_bollinger_bands(data, period=20, num_std=2):
    """Calculates Bollinger Bands."""
    sma = calculate_sma(data, period)
    std = pd.Series(data).rolling(window=period).std().values
    upper_band = sma + (std * num_std)
    lower_band = sma - (std * num_std)
    return upper_band, sma, lower_band

def ohlcv_to_dataframe(ohlcv_data):
    """
    Converts OHLCV data in string format to a Pandas DataFrame.

    Args:
        ohlcv_data (str): A string containing OHLCV data, where each row
                          represents a candlestick with Open, High, Low, Close, Volume values,
                          separated by commas. Rows are separated by newlines.

    Returns:
        pandas.DataFrame: A DataFrame with columns 'Open', 'High', 'Low', 'Close', 'Volume'.
                          Returns None if the input data is invalid or empty.
    """
    try:
        # Use StringIO to treat the string as a file
        data = StringIO(ohlcv_data)

        # Read the data into a DataFrame
        df = pd.read_csv(data, header=None, names=['Open', 'High', 'Low', 'Close', 'Volume'])

        # Return the DataFrame
        return df
    except Exception as e:
        print(f"Error converting OHLCV data to DataFrame: {e}")
        return None
