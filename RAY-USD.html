def calculate_rsi(prices, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = np.diff(prices)
    up, down = delta.copy(), delta.copy()
    up[up<0] = 0
    down[down>0] = 0
    
    roll_up1 = np.mean(up[:period])
    roll_down1 = np.mean(np.abs(down[:period]))

    down_list = list(np.abs(down[period:]))

    down_series = pd.Series(down_list)
    
    roll_up = up_series.ewm(alpha = 1/period).mean()
    roll_down = down_series.ewm(alpha = 1/period).mean()
    
    rs = roll_up / roll_down
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_sma(prices, period=20):
    """Calculates the Simple Moving Average (SMA)."""
    sma = np.mean(prices[-period:])
    return sma

def calculate_ema(prices, period=20):
    """Calculates the Exponential Moving Average (EMA)."""
    ema = pd.Series(prices).ewm(span=period, adjust=False).mean()
    return ema.iloc[-1]

def calculate_macd(prices, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    ema_fast = pd.Series(prices).ewm(span=fast_period, adjust=False).mean()
    ema_slow = pd.Series(prices).ewm(span=slow_period, adjust=False).mean()
    macd = ema_fast - ema_slow
    signal = macd.ewm(span=signal_period, adjust=False).mean()
    return macd.iloc[-1], signal.iloc[-1]

def calculate_bollinger_bands(prices, period=20, num_std=2):
  """Calculates Bollinger Bands."""
  sma = calculate_sma(prices, period)
  std = np.std(prices[-period:])
  upper_band = sma + num_std * std
  lower_band = sma - num_std * std
  return upper_band, lower_band
