def calculate_rsi(data, period=14): delta = data.diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = up.rolling(period).mean() roll_down1 = down.abs().rolling(period).mean() RS1 = roll_up1 / roll_down1 RSI1 = 100.0 - (100.0 / (1.0 + RS1)) return RSI1

def calculate_sma(data, period=20): sma = data.rolling(window=period).mean() return sma
