import numpy as np

def calculate_rsi(prices, period=14):
    delta = np.diff(prices)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = np.mean(up[:period])
    avg_loss = np.mean(np.abs(down[:period]))
    
    rs = avg_gain / avg_loss if avg_loss != 0 else 0
    rsi = 100 - (100 / (1 + rs))
    
    for i in range(period, len(prices) - 1):
        avg_gain = (avg_gain * (period - 1) + up[i]) / period
        avg_loss = (avg_loss * (period - 1) + np.abs(down[i])) / period
        rs = avg_gain / avg_loss if avg_loss != 0 else 0
        rsi = np.append(rsi, 100 - (100 / (1 + rs)))
    
    return rsi

def calculate_sma(prices, period=9):
    sma = np.mean(prices[:period])
    for i in range(period, len(prices)):
        sma = np.append(sma, np.mean(prices[i-period+1:i+1]))
    return sma
