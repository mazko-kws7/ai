# Simple Moving Average (SMA) - 20 days
def calculate_sma(data, period):
    return np.convolve(data, np.ones(period), 'valid') / period

sma_20 = calculate_sma(close_prices, 20)

# Relative Strength Index (RSI) - 14 days
def calculate_rsi(data, period=14):
    delta = np.diff(data)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    roll_up1 = np.convolve(up, np.ones(period), 'valid') / period
    roll_down1 = np.convolve(abs(down), np.ones(period), 'valid') / period
    
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

rsi_14 = calculate_rsi(close_prices)

# Moving Average Convergence Divergence (MACD)
def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    ema_fast = np.convolve(data, np.ones(fast_period), 'valid') / fast_period
    ema_slow = np.convolve(data, np.ones(slow_period), 'valid') / slow_period
    macd = ema_fast[slow_period-fast_period:] - ema_slow
    signal = np.convolve(macd, np.ones(signal_period), 'valid') / signal_period
    return macd, signal

macd, signal = calculate_macd(close_prices)
