import numpy as np

def calculate_sma(data, period):
    """Calculates the Simple Moving Average (SMA)."""
    if len(data) < period:
        return np.nan
    return np.mean(data[-period:])

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    if len(data) <= period:
        return np.nan

    deltas = np.diff(data)
    seed = deltas[:period+1]
    up = seed[seed >= 0].sum()/period
    down = -seed[seed < 0].sum()/period
    rs = up/down
    rsi = np.where(down == 0, 100, 100 - (100 / (1 + rs)))
    return rsi[-1]

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    if len(data) <= slow_period:
        return np.nan, np.nan, np.nan

    fast_ema = calculate_ema(data, fast_period)
    slow_ema = calculate_ema(data, slow_period)
    macd_line = fast_ema - slow_ema
    signal_line = calculate_ema(np.array(data[-signal_period:]), signal_period) # Using last signal_period data points for signal line calculation
    histogram = macd_line - signal_line

    return macd_line, signal_line, histogram

def calculate_ema(data, period):
    """Calculates the Exponential Moving Average (EMA)."""
    if len(data) < period:
        return np.nan

    ema = np.mean(data[:period])  # Seed value
    alpha = 2 / (period + 1)

    for i in range(period, len(data)):
        ema = alpha * data[i] + (1 - alpha) * ema

    return ema
