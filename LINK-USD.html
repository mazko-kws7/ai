df = pd.DataFrame(data, columns=['Open', 'High', 'Low', 'Close', 'Volume'])

# Calculate daily gains and losses
df['Change'] = df['Close'].diff()
df['Gain'] = df['Change'].apply(lambda x: x if x > 0 else 0)
df['Loss'] = df['Change'].apply(lambda x: abs(x) if x < 0 else 0)

# Calculate average gains and losses over 14 days
df['Avg_Gain'] = df['Gain'].rolling(window=14, min_periods=14).mean()
df['Avg_Loss'] = df['Loss'].rolling(window=14, min_periods=14).mean()

# Calculate RSI
df['RS'] = df['Avg_Gain'] / df['Avg_Loss']
df['RSI'] = 100 - (100 / (1 + df['RS']))

# Calculate 20-day SMA
df['SMA_20'] = df['Close'].rolling(window=20, min_periods=20).mean()

print(df[['Close', 'RSI', 'SMA_20']].tail(5))
