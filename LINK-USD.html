def calculate_rsi(prices, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given list of prices.

    Args:
      prices: A list of closing prices.
      period: The period over which to calculate the RSI (default: 14).

    Returns:
      A list of RSI values.
    """

    if len(prices) <= period:
        return []

    deltas = [prices[i] - prices[i-1] for i in range(1, len(prices))]
    seed = deltas[:period]
    up = sum(d for d in seed if d > 0) / period
    down = -sum(d for d in seed if d < 0) / period
    rs = up / down if down != 0 else 0  # Handle potential division by zero
    rsi = [100 - (100 / (1 + rs))]

    for i in range(period, len(deltas)):
        delta = deltas[i]
        up = (up * (period - 1) + (delta if delta > 0 else 0)) / period
        down = (down * (period - 1) + (-delta if delta < 0 else 0)) / period
        rs = up / down if down != 0 else 0 # Handle potential division by zero
        rsi.append(100 - (100 / (1 + rs)))

    return rsi
