import numpy as np
import pandas as pd

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = data.diff()
    delta = delta.dropna()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.rolling(period).mean()
    roll_down1 = down.abs().rolling(period).mean()
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

def calculate_sma(data, period=20):
    """Calculates the Simple Moving Average (SMA)."""
    return data.rolling(window=period).mean()

def calculate_ema(data, period=20):
    """Calculates the Exponential Moving Average (EMA)."""
    return data.ewm(span=period, adjust=False).mean()

def analyze_data(df, price_column='Close'):
    """
    Analyzes the given DataFrame for trading signals.

    Args:
    df (pd.DataFrame): DataFrame with OHLCV data.
    price_column (str): Name of the column to use for price analysis.

    Returns:
    pd.DataFrame: DataFrame with added technical indicators.
    """
    df['SMA_20'] = calculate_sma(df[price_column], period=20)
    df['EMA_20'] = calculate_ema(df[price_column], period=20)
    df['RSI'] = calculate_rsi(df[price_column], period=14)
    return df
