def calculate_ma(data, period):
    closes = [d[3] for d in data]
    ma = []
    for i in range(period, len(closes)):
        window = closes[i-period:i]
        ma.append(sum(window) / period)
    return ma

# Example: Calculate RSI (Relative Strength Index)
def calculate_rsi(data, period):
    closes = [d[3] for d in data]
    deltas = [closes[i] - closes[i-1] for i in range(1, len(closes))]
    seed = deltas[:period]
    up = sum([d for d in seed if d > 0])
    down = -sum([d for d in seed if d < 0])
    rs = up / down if down != 0 else 0
    rsi = 100 - (100 / (1 + rs))
    
    rsi_values = [rsi]
    
    for i in range(period, len(deltas)):
        delta = deltas[i]
        
        up_change = delta if delta > 0 else 0
        down_change = -delta if delta < 0 else 0
        
        up = (up * (period - 1) + up_change) / period
        down = (down * (period - 1) + down_change) / period
        
        rs = up / down if down != 0 else 0
        rsi = 100 - (100 / (1 + rs))
        rsi_values.append(rsi)
        
    return rsi_values
