def calculate_moving_average(data, period):
    """Calculates the simple moving average for a given period."""
    if len(data) < period:
        return None
    return sum(data[-period:]) / period

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    if len(data) < period:
        return None

    deltas = [data[i] - data[i - 1] for i in range(1, len(data))]
    if len(deltas) < period:
        return None

    avg_gain = sum([d for d in deltas[-period:] if d > 0]) / period
    avg_loss = abs(sum([d for d in deltas[-period:] if d < 0])) / period

    if avg_loss == 0:
        return 100 if avg_gain > 0 else 0

    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def analyze_trend(data, short_period, long_period):
    """Analyzes trend based on short and long period moving averages."""
    short_ma = calculate_moving_average(data, short_period)
    long_ma = calculate_moving_average(data, long_period)

    if short_ma is None or long_ma is None:
        return "Insufficient data"

    if short_ma > long_ma:
        return "Uptrend"
    elif short_ma < long_ma:
        return "Downtrend"
    else:
        return "Sideways"
