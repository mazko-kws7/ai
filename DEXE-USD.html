Technical Analysis Functions

import numpy as np
import pandas as pd

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    return pd.Series(data).rolling(window=period).mean()

def calculate_rsi(data, period=14):
    """Calculates Relative Strength Index."""
    delta = pd.Series(data).diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.ewm(alpha=1 / period).mean()
    roll_down1 = down.abs().ewm(alpha=1 / period).mean()
    RS1 = roll_up1 / roll_down1
    RSI1 = 100.0 - (100.0 / (1.0 + RS1))
    return RSI1

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """Calculates Moving Average Convergence Divergence."""
    ema_fast = pd.Series(data).ewm(span=fast_period, min_periods=fast_period).mean()
    ema_slow = pd.Series(data).ewm(span=slow_period, min_periods=slow_period).mean()
    macd = ema_fast - ema_slow
    signal = macd.ewm(span=signal_period, min_periods=signal_period).mean()
    histogram = macd - signal
    return macd, signal, histogram
