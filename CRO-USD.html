def calculate_sma(data, period):
    return np.convolve(data, np.ones(period), 'valid') / period

def calculate_rsi(data, period=14):
    delta = np.diff(data)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    roll_up = np.mean(up[-period:])
    roll_down = -np.mean(down[-period:])
    rs = roll_up / roll_down if roll_down != 0 else 0
    rsi = 100 - (100 / (1 + rs))
    return rsi
