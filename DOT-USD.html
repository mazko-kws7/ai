import pandas as pd
import numpy as np

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    return data['Close'].rolling(window=period).mean()

def calculate_rsi(data, period):
    """Calculates Relative Strength Index."""
    delta = data['Close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = up.rolling(window=period).mean()
    avg_loss = abs(down.rolling(window=period).mean())
    
    rs = avg_gain / avg_loss
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_macd(data, fast_period, slow_period, signal_period):
    """Calculates Moving Average Convergence Divergence."""
    ema_fast = data['Close'].ewm(span=fast_period, adjust=False).mean()
    ema_slow = data['Close'].ewm(span=slow_period, adjust=False).mean()
    macd = ema_fast - ema_slow
    signal = macd.ewm(span=signal_period, adjust=False).mean()
    return macd, signal

def analyze_price_action(data):
  """
  Analyzes price action to identify potential bullish or bearish signals based on candlestick patterns.
  This is a simplified analysis and can be expanded for more robust pattern recognition.
  """
  if len(data) < 2:
    return "Neutral"

  current_close = data['Close'].iloc[-1]
  previous_close = data['Close'].iloc[-2]
  current_open = data['Open'].iloc[-1]
  previous_open = data['Open'].iloc[-2]

  # Bullish engulfing pattern
  if current_open < previous_close and current_close > previous_open and current_close > current_open:
    return "Bullish"

  # Bearish engulfing pattern
  if current_open > previous_close and current_close < previous_open and current_close < current_open:
    return "Bearish"

  # Doji or spinning top (Indecision)
  if abs(current_close - current_open) < (data['High'].iloc[-1] - data['Low'].iloc[-1]) * 0.1:  # Small body
    return "Neutral"

  # If none of the above, check for simple up or down day
  if current_close > current_open:
    return "Slightly Bullish"
  elif current_close < current_open:
    return "Slightly Bearish"
  else:
    return "Neutral"
