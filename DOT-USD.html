# Calculate RSI (14-day)
def calculate_rsi(data, window=14):
    delta = data.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.rolling(window).mean()
    roll_down1 = down.abs().rolling(window).mean()
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

daily_df['RSI'] = calculate_rsi(daily_df['Close'])

# Calculate MACD (12, 26, 9)
EMA_12 = daily_df['Close'].ewm(span=12, adjust=False).mean()
EMA_26 = daily_df['Close'].ewm(span=26, adjust=False).mean()
daily_df['MACD'] = EMA_12 - EMA_26
daily_df['Signal'] = daily_df['MACD'].ewm(span=9, adjust=False).mean()
