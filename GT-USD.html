def calculate_rsi(data, period=14): ""Calculates the Relative Strength Index (RSI)."" delta = data.diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = up.ewm(span=period, adjust=False).mean() roll_down1 = down.abs().ewm(span=period, adjust=False).mean() RS = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS)) return RSI

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9): ""Calculates the Moving Average Convergence Divergence (MACD)."" EMA_fast = data.ewm(span=fast_period, adjust=False).mean() EMA_slow = data.ewm(span=slow_period, adjust=False).mean() MACD = EMA_fast - EMA_slow signal = MACD.ewm(span=signal_period, adjust=False).mean() histogram = MACD - signal return MACD, signal, histogram

def calculate_sma(data, period=20): ""Calculates the Simple Moving Average (SMA)."" return data.rolling(window=period).mean()
