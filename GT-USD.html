def calculate_rsi(closes, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    gains = []
    losses = []
    for i in range(1, len(closes)):
        change = closes[i] - closes[i - 1]
        if change > 0:
            gains.append(change)
            losses.append(0)
        else:
            gains.append(0)
            losses.append(abs(change))

    avg_gain = sum(gains[:period]) / period
    avg_loss = sum(losses[:period]) / period

    for i in range(period, len(gains)):
        avg_gain = (avg_gain * (period - 1) + gains[i]) / period
        avg_loss = (avg_loss * (period - 1) + losses[i]) / period

    rs = avg_gain / avg_loss if avg_loss != 0 else 0
    rsi = 100 - (100 / (1 + rs))
    return rsi

def calculate_macd(closes, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    fast_ema = calculate_ema(closes, fast_period)
    slow_ema = calculate_ema(closes, slow_period)
    macd_line = [fast_ema[i] - slow_ema[i] for i in range(len(fast_ema))]
    signal_line = calculate_ema(macd_line, signal_period)
    return macd_line[-1] if macd_line else None, signal_line[-1] if signal_line else None

def calculate_ema(closes, period):
  ema = []
  k = 2 / (period + 1)
  ema.append(sum(closes[:period]) / period)
  for i in range(period, len(closes)):
    ema.append(closes[i] * k + ema[-1] * (1 - k))
  return ema

def analyze_trend(closes):
    """Analyzes the trend based on moving averages."""
    short_period = 5
    long_period = 20

    if len(closes) < long_period:
        return "Insufficient Data"

    short_ma = sum(closes[-short_period:]) / short_period
    long_ma = sum(closes[-long_period:]) / long_period

    if short_ma > long_ma:
        return "Uptrend"
    elif short_ma < long_ma:
        return "Downtrend"
    else:
        return "Sideways"
