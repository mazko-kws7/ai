def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    if len(data) < period:
        return None
    return sum(data[-period:]) / period

def calculate_rsi(data, period=14):
    """Calculates Relative Strength Index."""
    if len(data) <= period:
        return None

    # Calculate gains and losses
    gains = []
    losses = []
    for i in range(1, len(data)):
        change = data[i] - data[i-1]
        if change > 0:
            gains.append(change)
        else:
            losses.append(abs(change))

    # Calculate average gains and losses for the first 'period'
    avg_gain = sum(gains[:period]) / period if len(gains) >= period else 0
    avg_loss = sum(losses[:period]) / period if len(losses) >= period else 0

    # Calculate RS and RSI
    if avg_loss == 0:
        return 100  # To avoid division by zero if there are no losses

    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def analyze_volume_spikes(volumes):
    """Identifies significant volume increases."""
    if len(volumes) < 2:
        return False

    recent_volume = volumes[-1]
    average_volume = sum(volumes[:-1]) / (len(volumes) - 1)
    
    #Detect volume spike
    volume_spike_threshold = 2
    if recent_volume > volume_spike_threshold * average_volume:
        return True
    return False
