        entry_match = re.search(r"Entry Price:\s*([\d.]+)", content)

        if recommendation_match and entry_match and stop_loss_match and probability_match:
            recommendation = recommendation_match.group(1)

            stop_loss = float(stop_loss_match.group(1))
            probability = float(probability_match.group(1))

            return {
                "recommendation": recommendation,
                "entry_price": entry_price,
                "stop_loss": stop_loss,
                "probability": probability
            }
        else:
            print(f"Could not parse data from {filename}")
            return None  # Or raise an exception
    except FileNotFoundError:
        print(f"File not found: {filename}")
        return None
