import re

def summarize_reports(reports_text):
    """
    Parses trading reports, extracts asset name and probability, sorts by probability,
    and prints a summary.

    Args:
        reports_text: A string containing all reports, separated by lines like
                      '==> response/ASSET_NAME.md <==\n report content\n...'
    """

    reports = reports_text.strip().split("==> ")
    reports_data = []

    for report_block in reports:
        if not report_block: # Skip empty string
            continue

        try:
            filename_line, content = report_block.split("\n", 1) # split into filename line and actual report
            match = re.match(r"response/(.+)\.md <==", filename_line.strip())

            if match:
                asset_name = match.group(1)

                # Very simple probability extraction (improve if needed)
                probability_match = re.search(r"probability:\s*([0-9.]+)", content) # search for pattern
                if probability_match:
                    try:
                        probability = float(probability_match.group(1))
                    except ValueError:
                        probability = None # or a default value

                    reports_data.append({"asset": asset_name, "probability": probability})

        except ValueError as e:
            print(f"Error processing report block: {e}")
            print(f"Report block: {report_block}")


    # Sort by probability (descending)
    reports_data.sort(key=lambda x: x["probability"] if x["probability"] is not None else -1, reverse=True)

    print("Report Summary (Sorted by Probability):")
    for report in reports_data:
        print(f"  Asset: {report['asset']}, Probability: {report['probability']}")
