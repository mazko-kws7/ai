import glob
import re

def analyze_report(filename, file_content):
    """Extracts information from a report and returns a dictionary."""
    asset_name = filename.split('/')[1].split('.')[0] #e.g., AAPL
    # Use regular expressions to find the probability (adjust the regex as needed)
    match = re.search(r"Profit Probability: (\d+)%", file_content)
    if match:
        probability = int(match.group(1))
    else:
        probability = 0  # Or handle the error in a different way

    return {"asset": asset_name, "probability": probability}

# Get list of files
files = glob.glob("response/*.md")

reports = []

for filename in files:
    with open(filename, 'r') as f:
        content = f.read()
        report_data = analyze_report(filename, content)
        reports.append(report_data)

# Sort the reports by probability (descending)
reports.sort(key=lambda x: x["probability"], reverse=True)

# Print the summary
for report in reports:
    print(f"ASSET: {report['asset']}, Probability: {report['probability']}%")
