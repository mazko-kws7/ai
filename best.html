import re

report_data = []

def analyze_report(filename, report_content):
    """
    Analyzes the report content to extract relevant information.

    Args:
        filename (str): The filename of the report (e.g., "response/AAPL.md").
        report_content (str): The content of the report file.

    Returns:
        dict: A dictionary containing the extracted data, or None if extraction fails.
    """
    asset_name = filename.split('/')[1].split('.')[0]  # Extract ASSET_NAME from filename
    # Example regex (adjust as needed based on your report format):
    match = re.search(r"Recommendation:\s*(LONG|SHORT)\s*Entry Price:\s*([\d.]+)\s*Stop Loss:\s*([\d.]+)\s*Profit Probability:\s*([\d.]+)%", report_content)
    if match:
        recommendation, entry_price, stop_loss, profit_probability = match.groups()
        return {
            "Asset": asset_name,
            "Recommendation": recommendation,
            "Entry Price": float(entry_price),
            "Stop Loss": float(stop_loss),
            "Profit Probability": float(profit_probability) / 100.0 #convert % to decimal
        }
    else:
        print(f"Could not parse report for {filename}")
        return None

def add_report(filename, report_content):
    """
    Adds a new report to the list of reports to be analyzed.
    """
    global report_data
    data = analyze_report(filename, report_content)
    if data:
        report_data.append(data)


def generate_summary():
    """
    Generates a sorted summary of the reports based on profit probability.
    """
    global report_data
    # Sort the data by profit probability in descending order
    sorted_data = sorted(report_data, key=lambda x: x["Profit Probability"], reverse=True)

    print("Summary (Sorted by Profit Probability):")
    for item in sorted_data:
        print(f"- {item['Asset']}: {item['Profit Probability']:.2f}")
