import random

def analyze_report(asset_name):
    """Simulates reading a report and extracting data."""
    recommendation = random.choice(["LONG", "SHORT"])
    entry_price = round(random.uniform(50, 200), 2)
    stop_loss = round(entry_price * (1 + random.uniform(-0.1, 0.05) if recommendation == "SHORT" else random.uniform(-0.05, 0.1)), 2) #make SL based on LONG/SHORT
    probability = round(random.uniform(0.5, 0.95), 2) # Profit probability between 50% and 95%
    return {
        "asset_name": asset_name,
        "recommendation": recommendation,
        "entry_price": entry_price,
        "stop_loss": stop_loss,
        "probability": probability
    }

def generate_reports(asset_names):
    """Generates simulated reports for a list of asset names."""
    reports = []
    for asset_name in asset_names:
        reports.append(analyze_report(asset_name))
    return reports

def summarize_and_sort(reports):
    """Summarizes reports and sorts them by profit probability."""
    summaries = []
    for report in reports:
        summaries.append({
            "asset_name": report["asset_name"],
            "probability": report["probability"]
        })

    sorted_summaries = sorted(summaries, key=lambda x: x["probability"], reverse=True)
    return sorted_summaries
