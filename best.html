# Function to extract profit probability
def extract_probability(report_content):
  match = re.search(r"Profit Probability: (\d+)%", report_content)
  if match:
    return int(match.group(1))
  else:
    return None # Handle cases where the probability is not found

# Analyze each report and store the data
analysis_results = []
for asset, content in reports.items():
  probability = extract_probability(content)
  if probability is not None:
      analysis_results.append({"asset": asset, "probability": probability})
  else:
      print(f"Warning: Could not extract probability from {asset}.  Skipping.")


# Sort by profit probability in descending order
sorted_results = sorted(analysis_results, key=lambda x: x["probability"], reverse=True)

# Print the sorted resume
print("Trading Recommendation Resume (Sorted by Profit Probability):")
for result in sorted_results:
  print(f"Asset: {result['asset']}, Profit Probability: {result['probability']}%")
