import os
import re

def analyze_reports(directory="response"):  # Changed to directory instead of wildcard
    """
    Analyzes trading reports in markdown files within a directory, extracts
    relevant information, sorts them by take profit probability, and
    prints a summary.
    """

    reports = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename.replace("ASSET_", "").replace(".md", "").split('/')[-1]  # Extract asset name
            filepath = os.path.join(directory, filename)
            try:
                with open(filepath, 'r') as f:
                    content = f.read()

                    # Extract relevant information using regular expressions.  Robust error handling is crucial.
                    position_match = re.search(r"POSITION:\s*(SHORT|LONG)", content, re.IGNORECASE)
                    probability_match = re.search(r"PROBABILITY:\s*([0-9.]+)", content, re.IGNORECASE) # changed to probability

                    if position_match and probability_match:
                        position = position_match.group(1).upper()
                        probability = float(probability_match.group(1))
                        reports.append({
                            'asset_name': asset_name,
                            'position': position,
                            'probability': probability,
                            'filename': filepath
                        })
                    else:
                        print(f"Warning: Could not extract all information from {filename}. Skipping.")

            except FileNotFoundError:
                print(f"Error: File not found: {filename}")
            except Exception as e:
                print(f"Error reading {filename}: {e}")

    # Sort reports by probability (highest first)
    reports.sort(key=lambda x: x['probability'], reverse=True)

    # Print the summary
    print("Trading Report Summary (Sorted by Probability):")
    for report in reports:
        print(f"Asset: {report['asset_name']}, Position: {report['position']}, Probability: {report['probability']:.2f}, File: {report['filename']}")

How to Adapt to Your Specific Report Format

The script relies on regular expressions to extract the data from your reports. If your report format is different, you’ll need to modify the regular expressions accordingly. Here’s how:

  1. Examine a Report: Open one of your ASSET_NAME.md files and carefully look at the structure.
  2. Identify Patterns: Find the consistent patterns that identify the “POSITION,” and “PROBABILITY” values.
