Example (Python):

import glob
import re  # for regular expressions (more robust parsing)

def analyze_reports(directory="response"):
    """
    Analyzes trading reports in a directory and returns a sorted summary.
    """

    reports_data = []

    for filepath in glob.glob(f"{directory}/*.md"):
        try:
            with open(filepath, 'r') as f:
                report_content = f.read()

            # Extract asset name from filename
            asset_name = filepath.split('/')[-1].replace(".md", "")  #Assumes directory is named "response"

            # Use regular expressions for more robust parsing:
            recommendation_match = re.search(r"Recommendation:\s*(SHORT|LONG)", report_content)
            entry_price_match = re.search(r"Entry Price:\s*([\d.]+)", report_content)
            stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", report_content)
            probability_match = re.search(r"Profit Probability:\s*([\d.]+)%", report_content)

            if recommendation_match and entry_price_match and stop_loss_match and probability_match:
                recommendation = recommendation_match.group(1)
                entry_price = float(entry_price_match.group(1))
                stop_loss = float(stop_loss_match.group(1))
                profit_probability = float(probability_match.group(1))

                report_data = {
                    'asset_name': asset_name,
                    'recommendation': recommendation,
                    'entry_price': entry_price,
                    'stop_loss': stop_loss,
                    'profit_probability': profit_probability
                }
                reports_data.append(report_data)
            else:
                print(f"Warning: Could not parse all information from {filepath}")

        except FileNotFoundError:
            print(f"Error: File not found: {filepath}")
        except Exception as e:
            print(f"Error processing {filepath}: {e}")


    # Sort by profit probability (descending)
    reports_data.sort(key=lambda x: x['profit_probability'], reverse=True)

    summary = ""
    for report in reports_data:
        summary += f"- {report['asset_name']}: Probability = {report['profit_probability']:.2f}%\n"  #Format to 2 decimal places


    return summary

# Example Usage (assuming the files are in a directory named "response"):
summary = analyze_reports("response")
print(summary)
