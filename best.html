import os
import re  # For regular expressions (if needed for parsing)

def parse_report(filename):
    """Parses the report file and returns a dictionary of information."""
    try:
        with open(filename, 'r') as f:
            content = f.read()

        # Extract information using parsing logic (example using regex)
        position_match = re.search(r"Position:\s*(\w+)", content)
        entry_match = re.search(r"Entry:\s*([\d.]+)", content)
        stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content)
        probability_match = re.search(r"Probability:\s*([\d.]+)", content)

        if position_match and entry_match and stop_loss_match and probability_match:
            position = position_match.group(1)
            entry = float(entry_match.group(1))
            stop_loss = float(stop_loss_match.group(1))
            probability = float(probability_match.group(1))
        else:
            print(f"Warning: Could not parse all fields in {filename}")
            return None


        return {
            'asset': os.path.splitext(os.path.basename(filename))[0].split('_')[1],  # Extract ASSET_NAME
            'position': position,
            'entry': entry,
            'stop_loss': stop_loss,
            'probability': probability
        }
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return None
    except Exception as e:
        print(f"Error processing {filename}: {e}")
        return None


def generate_summary(report):
    """Generates a summary string for a report."""
    return f"{report['asset']}: {report['probability']:.0%} Profit Probability"  #Format to percent.

def main(report_files):
    """Main function to process the reports and print the sorted summary."""
    reports = []
    for filename in report_files:
      report = parse_report(filename)
      if report:
        reports.append(report)


    reports.sort(key=lambda x: x['probability'], reverse=True)

    for report in reports:
        print(generate_summary(report))
