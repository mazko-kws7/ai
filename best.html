import os
import re

def analyze_reports(report_directory="response"):
    """
    Analyzes trading reports, extracts asset name, and profit probability,
    and sorts them by probability in descending order.

    Args:
        report_directory: The directory containing the report files.

    Returns:
        A list of tuples, where each tuple contains (asset_name, probability),
        sorted by probability in descending order.  Returns an empty list
        if no reports are found or an error occurs.
    """

    report_data = []
    try:
        for filename in os.listdir(report_directory):
            if filename.endswith(".md"):  # or whatever your file extension is
                filepath = os.path.join(report_directory, filename)
                asset_name = filename.replace(".md", "") # Extract asset name from filename

                with open(filepath, "r") as f:
                    file_content = f.read()

                # **EXTRACT DATA FROM FILE CONTENT**  (This is the key part you'll need to adapt)
                # Example using regular expressions:

                probability_match = re.search(r"Probability:\s*(\d+\.?\d*)", file_content)

                if probability_match:
                    probability = float(probability_match.group(1))

                    # **Extract other information (Position, Entry Price, Stop Loss) here if you need them**

                    report_data.append((asset_name, probability))
                else:
                    print(f"Warning: Could not find Probability in {filename}")

    except FileNotFoundError:
        print(f"Error: Directory '{report_directory}' not found.")
        return []
    except Exception as e:
        print(f"Error processing reports: {e}")
        return []

    # Sort by probability in descending order
    sorted_reports = sorted(report_data, key=lambda item: item[1], reverse=True)
    return sorted_reports


def main():
    report_summaries = analyze_reports()

    if report_summaries:
        print("Report Summaries (Sorted by Take Profit Probability):")
        for asset_name, probability in report_summaries:
            print(f"  Asset: {asset_name}, Probability: {probability:.2f}")  # Format probability to 2 decimal places
    else:
        print("No reports found or an error occurred.")


if __name__ == "__main__":
