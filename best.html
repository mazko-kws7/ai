#!/bin/bash

# Array to store the data for sorting
declare -a reports

# Loop through the files
for file in response/*.md; do
  if [ -f "$file" ]; then
    # Extract asset name from filename
    asset_name=$(basename "$file" | sed 's/response\/\(.*\)\.md/\1/')

    # Extract profit probability (example using grep and awk - adjust as needed)
    profit_probability=$(grep "profit probability" "$file" | awk '{print $NF}' | tr -d '%')

    # Add data to the reports array
    reports+=("$asset_name|$profit_probability")

  fi
done

# Function to sort the array based on the second field (profit probability)
sort_by_probability() {
  echo "$@" | tr ' ' '\n' | sort -t'|' -k2,2nr
}

# Sort the array
sorted_reports=$(sort_by_probability "${reports[@]}")

# Print the sorted output
while read -r report; do
  asset_name=$(echo "$report" | cut -d'|' -f1)
  probability=$(echo "$report" | cut -d'|' -f2)
  echo "Asset: $asset_name, Probability: $probability%"
done <<< "$sorted_reports"
