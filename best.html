import re

reports = []

        asset = filename[:-3]  # Remove ".md"
        filepath = os.path.join("response", filename)
        try:
            with open(filepath, "r") as f:
                content = f.read()
                match = re.search(r"Profit probability: (\d+)%", content)
                if match:
                    probability = int(match.group(1))
                    reports.append({"asset": asset, "probability": probability})

        except FileNotFoundError:
            print(f"Error: File not found: {filename}")
        except Exception as e:
            print(f"Error processing {filename}: {e}")

reports.sort(key=lambda x: x['probability'], reverse=True)

for report in reports:
