import os
import re

def process_report(filename):
    """Processes a single report file."""
    try:
        with open(filename, 'r') as f:
            content = f.read()

        # Extract information using regular expressions (adapt as needed)
        position_match = re.search(r"(LONG|SHORT)", content)
        entry_match = re.search(r"(Entry Price:|Entry:)\s*([\d.]+)", content)
        stop_match = re.search(r"(Stop Loss:|Stop:)\s*([\d.]+)", content)
        probability_match = re.search(r"(Profit Probability:|Probability:)\s*([\d.]+)%", content)

        if position_match and entry_match and stop_match and probability_match:
            position = position_match.group(1)
            entry_price = float(entry_match.group(2))
            stop_loss = float(stop_match.group(2))
            probability = float(probability_match.group(2))
            return {
                "asset": os.path.splitext(os.path.basename(filename))[0],  # Extract asset name from filename
                "position": position,
                "entry_price": entry_price,
                "stop_loss": stop_loss,
                "probability": probability
            }
        else:
            print(f"Warning: Could not parse all information from {filename}")
            return None
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return None

def main():
    """Main function to process all report files."""
    report_files = [f"response/{filename.split('/')[1]}" for filename in open('list_of_files.txt', 'r')] # create list of files from your input
    # report_files = [f"response/{f}" for f in os.listdir("response") if f.endswith(".md")] #Use if you have the response files in a directory named response

    reports = []
    for file in report_files:
        report_data = process_report(file)
        if report_data:
            reports.append(report_data)

    # Sort by probability (highest to lowest)
    sorted_reports = sorted(reports, key=lambda x: x["probability"], reverse=True)

    # Print the sorted summaries
    for report in sorted_reports:
        print(f"Asset: {report['asset']}, Profit Probability: {report['probability']:.2f}%")

if __name__ == "__main__":
    main()
