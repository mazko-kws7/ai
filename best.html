How to Implement (Conceptual):

You’ll need to use a scripting language (like Python, Bash, or similar) to automate this process. Here’s a very high-level outline in Python:

import os
import re

reports = []

#  Assume files are in a directory named 'response'
for filename in os.listdir('response'):
    if filename.endswith(".md"):
        asset_name = filename[:-3]  # Remove ".md"
        filepath = os.path.join('response', filename)

        with open(filepath, 'r') as f:
            content = f.read()

            # Use regular expressions to extract the probability and recommendation
            probability_match = re.search(r"Probability.* (\d+\.?\d*)%", content) #Example: Probability of success is 95.5%.
            recommendation_match = re.search(r"Recommendation:\s*([A-Z]+)", content)

            if probability_match and recommendation_match:
                probability = float(probability_match.group(1))
                recommendation = recommendation_match.group(1)
                reports.append((asset_name, probability, recommendation))
            else:
                print(f"Warning: Could not parse {filename}")


# Sort by probability in descending order
reports.sort(key=lambda x: x[1], reverse=True)


for asset_name, probability, recommendation in reports:
    print(f"{asset_name}: Probability - {probability:.1f}%, Recommendation - {recommendation}")
