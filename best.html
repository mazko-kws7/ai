    import glob
import re

reports = []
for filename in glob.glob("response/*.md"):
    with open(filename, "r") as f:
        content = f.read()
        match = re.search(r"Profit Probability: (\d+)%", content)
        if match:
            probability = int(match.group(1))
            asset_name = filename.replace("response/", "").replace(".md", "")
            reports.append((asset_name, probability))

reports.sort(key=lambda x: x[1], reverse=True)

for asset_name, probability in reports:
    print(f"{asset_name}: Probability: {probability}%")
