import glob
import re

def extract_data(filepath):
    """Extracts trading data from a report file."""
    asset_name = filepath.split('/')[1].split('.')[0]  # Assuming format 'response/ASSET_NAME.md'
    with open(filepath, 'r') as f:
        content = f.read()

    #  Use regular expressions or other parsing methods to extract the data.
    #  Adjust these patterns based on the actual content of your .md files!
    recommendation_match = re.search(r"Recommendation:\s*(LONG|SHORT)", content)
    entry_match = re.search(r"Entry Price:\s*([\d.]+)", content)
    stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content)
    probability_match = re.search(r"Probability:\s*([\d.]+)%", content)


    recommendation = recommendation_match.group(1) if recommendation_match else "N/A"
    entry_price = float(entry_match.group(1)) if entry_match else None
    stop_loss = float(stop_loss_match.group(1)) if stop_loss_match else None
    probability = float(probability_match.group(1)) if probability_match else None


    return {
        "asset_name": asset_name,
        "recommendation": recommendation,
        "entry_price": entry_price,
        "stop_loss": stop_loss,
        "probability": probability
    }

# Get list of files
file_pattern = 'response/*.md'
report_files = glob.glob(file_pattern)
if not report_files:
  print("No files found matching the pattern:", file_pattern)
  exit()

# Extract data from each file
data = []
for filepath in report_files:
    try:
        data.append(extract_data(filepath))
    except Exception as e:
        print(f"Error processing file {filepath}: {e}")

# Remove items where probability is None
data = [item for item in data if item['probability'] is not None]

# Sort by probability
sorted_data = sorted(data, key=lambda x: x["probability"])


print("Summary of Trading Recommendations (Sorted by Take Profit Probability):")
for item in sorted_data:
    print(f"- **Asset:** {item['asset_name']}, **Probability:** {item['probability']}% (Recommendation: {item['recommendation']}, Entry: {item['entry_price']}, Stop Loss: {item['stop_loss']})")
