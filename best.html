import pandas as pd import numpy as np

def analyze_crypto(daily_data, hourly_data): ""“Analyzes crypto data and returns trading signals.”""

# Convert data to pandas DataFrames
daily_df = pd.read_csv(io.StringIO(daily_data), header=None, names=['Open', 'High', 'Low', 'Close', 'Volume'])
hourly_df = pd.read_csv(io.StringIO(hourly_data), header=None, names=['Open', 'High', 'Low', 'Close', 'Volume'])

# --- Calculate Technical Indicators (Daily) ---
def calculate_sma(data, period):
    return data['Close'].rolling(window=period).mean()

def calculate_rsi(data, period=14):
    delta = data['Close'].diff()
    up, down = delta.clip(lower=0), -delta.clip(lower=0)
    roll_up1 = up.ewm(com=(period - 1), adjust=False).mean()
    roll_down1 = down.ewm(com=(period - 1), adjust=False).mean()
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    EMA_fast = data['Close'].ewm(span=fast_period, adjust=False).mean()
    EMA_slow = data['Close'].ewm(span=slow_period, adjust=False).mean()
    MACD = EMA_fast - EMA_slow
    Signal = MACD.ewm(span=signal_period, adjust=False).mean()
    return MACD, Signal

daily_df['SMA_50'] = calculate_sma(daily_df, 50)
daily_df['SMA_200'] = calculate_sma(daily_df, 200)
daily_df['RSI'] = calculate_rsi(daily_df)
daily_df['MACD'], daily_df['Signal'] = calculate_macd(daily_df)

# Remove rows with NaN values resulting from calculations
daily_df = daily_df.dropna()

# --- Generate Trading Signals ---
signals = []

# Signal 1: Buy if RSI is oversold and MACD is crossing over
if daily_df['RSI'].iloc[-1] < 30 and daily_df['MACD'].iloc[-1] > daily_df['Signal'].iloc[-1]:
    signals.append("""

XRP-USD

  • Buy 65%

  • The RSI is oversold, suggesting potential for a rebound. MACD crossover confirms upward momentum. """)

    # Signal 2: Sell if RSI is overbought and MACD is crossing over if daily_df[‘RSI’].iloc[-1] > 70 and daily_df[‘MACD’].iloc[-1] < daily_df[‘Signal’].iloc[-1]: signals.append(""" ## XRP-USD

  • Sell 80%

  • The RSI is overbought, suggesting potential for a pullback. MACD crossover confirms downward momentum. """)

    # Signal 3: Buy if price is above both SMAs if daily_df[‘Close’].iloc[-1] > daily_df[‘SMA_50’].iloc[-1] and daily_df[‘Close’].iloc[-1] > daily_df[‘SMA_200’].iloc[-1]: signals.append(""" ## XRP-USD

  • Buy 70%

  • The price is above both the 50-day and 200-day SMAs, indicating a strong upward trend. """)

    # Signal 4: Sell if price is below both SMAs if daily_df[‘Close’].iloc[-1] < daily_df[‘SMA_50’].iloc[-1] and daily_df[‘Close’].iloc[-1] < daily_df[‘SMA_200’].iloc[-1]: signals.append(""" ## XRP-USD

  • Sell 90%

  • The price is below both the 50-day and 200-day SMAs, indicating a strong downward trend. """)

    # Sort signals by take profit probability (extract the probability from string, convert to int, and sort) signals.sort(key=lambda x: int(x.split(‘-’)[1].split(‘%’)[0]), reverse=True)

    return “”.join(signals)
