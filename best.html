import os
import re

def process_reports(directory="response"):
    """
    Processes trading reports, extracts information, and sorts by profit probability.
    """

    reports = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[:-3]  # Remove ".md"
            filepath = os.path.join(directory, filename)

            try:
                with open(filepath, "r") as f:
                    content = f.read()

                    # Use regular expressions to extract relevant data.
                    # You'll need to adapt these regexes to your specific file format.
                    position_match = re.search(r"Position:\s*(LONG|SHORT)", content)
                    probability_match = re.search(r"Probability:\s*(\d+)%", content)  # Assuming probability is in %

                    if position_match and probability_match:
                        position = position_match.group(1)
                        probability = int(probability_match.group(1))

                        reports.append({
                            "asset": asset_name,
                            "position": position,
                            "probability": probability
                        })
                    else:
                        print(f"Warning: Could not extract data from {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")

    # Sort reports by probability in descending order
    sorted_reports = sorted(reports, key=lambda x: x["probability"], reverse=True)

    # Print the sorted summaries
    for report in sorted_reports:
        print(f"Asset: {report['asset']}, Position: {report['position']}, Probability: {report['probability']}%")


# Example usage (assuming reports are in the 'response' directory)
process_reports()
