import os
import re

def extract_report_data(filename):
    """Extracts asset name and profit probability from a report file."""
    asset_name = filename.split('/')[1].split('.')[0]  # e.g., "AAPL" from "response/AAPL.md"
    try:
        with open(filename, 'r') as f:
            content = f.read()
            match = re.search(r"Profit Probability: (\d+)%", content)
            if match:
                probability = int(match.group(1))
                return asset_name, probability
            else:
                print(f"Warning: Could not find profit probability in {filename}")
                return asset_name, -1  # Or None, or some other sentinel value

    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return asset_name, -1

def main():
    """Main function to process reports and print the sorted summary."""
    report_data = []

    # Replace with your actual path if needed or use relative path if the script is
    # in the same directory
    directory = "response"  # Relative directory
    if not os.path.exists(directory):
        print(f"Error: Directory '{directory}' not found.")
        return

    for filename in os.listdir(directory):
        if filename.endswith(".md"):  # Only process Markdown files
            filepath = os.path.join(directory, filename)
            asset_name, probability = extract_report_data(filepath)
            if probability != -1: # filter out files for which we could not determine probability
                report_data.append((asset_name, probability))

    # Sort by probability in descending order
    report_data.sort(key=lambda item: item[1], reverse=True)

    # Print the sorted summary
    for asset_name, probability in report_data:
        print(f"{asset_name}: {probability}%")

if __name__ == "__main__":
    main()

  • Function Decomposition: The code is broken down into functions (extract_report_data and main) for better readability and organization.
