import re

def extract_report_data(filename):
    """Extracts trading data from a report file."""
    try:
        with open(filename, 'r') as f:
            content = f.read()
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return None

    # Extract position (LONG or SHORT)
    position_match = re.search(r"(LONG|SHORT)", content, re.IGNORECASE)
    position = position_match.group(1).upper() if position_match else "UNKNOWN"

    # Extract Entry Price
    entry_match = re.search(r"Entry Price:\s*([\d.]+)", content, re.IGNORECASE)
    entry_price = float(entry_match.group(1)) if entry_match else None

    # Extract Stop Loss
    stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content, re.IGNORECASE)
    stop_loss = float(stop_loss_match.group(1)) if stop_loss_match else None

    # Extract Profit Probability
    probability_match = re.search(r"Profit Probability:\s*([\d.]+)%", content, re.IGNORECASE)
    probability = float(probability_match.group(1)) if probability_match else None

    return {
        "position": position,
        "entry_price": entry_price,
        "stop_loss": stop_loss,
        "probability": probability
    }

def summarize_reports(report_list):
    """Summarizes reports and sorts by profit probability."""
    report_summaries = []
    for report_path in report_list:
        asset_name = report_path.split('/')[1].split('.')[0]  # Extract ASSET_NAME
        data = extract_report_data(report_path)
        if data:
            if data["probability"] is not None:
                report_summaries.append({
                    "asset": asset_name,
                    "probability": data["probability"],
                    "position": data["position"],
                    "entry_price": data["entry_price"],
                    "stop_loss": data["stop_loss"]
                })
            else:
                print(f"Warning: Could not extract probability from {report_path}")
        else:
            print(f"Warning: Could not process {report_path}")
    # Sort by probability in descending order
    report_summaries.sort(key=lambda x: x["probability"], reverse=True)
    return report_summaries
