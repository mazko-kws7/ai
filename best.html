import os
import re  # For regular expressions (adjust to your file format)

def analyze_reports(directory="response"):
    """
    Analyzes trading reports in the specified directory, extracts
    asset name and profit probability, and sorts them by probability.
    """

    report_data = []

    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[:-3]  # Remove ".md" extension

            try:
                with open(os.path.join(directory, filename), "r") as f:
                    content = f.read()

                    # Use regex to extract the Profit Probability and Recommendation
                    probability_match = re.search(r"Profit Probability:\s*(\d+)%", content)
                    recommendation_match = re.search(r"(SHORT|LONG)", content)  #Find SHORT OR LONG recomendation

                    if probability_match and recommendation_match:
                        probability = int(probability_match.group(1))
                        recommendation = recommendation_match.group(1)

                        report_data.append(
                            {"asset": asset_name, "probability": probability, "recommendation":recommendation}
                        )
                    else:
                        print(f"Warning: Could not extract probability or recommendation from {filename}")


            except FileNotFoundError:
                print(f"Error: File not found: {filename}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")


    # Sort the list by probability in descending order
    report_data.sort(key=lambda x: x["probability"], reverse=True)

    print("Summary of Trading Recommendations (Sorted by Profit Probability):")
    for item in report_data:
        print(f"ASSET_NAME: {item['asset']}, Recommendation: {item['recommendation']}, Profit Probability: {item['probability']}%")

# Example usage (assuming the 'response' directory exists and contains files)
analyze_reports()
