import re

          content = f.read()
          #Use RegEx to find required data
          #This is sample - depends on real file structure
          recommendation_match = re.search(r"Recommendation:\s*(LONG|SHORT)", content)
          probability_match = re.search(r"Probability:\s*(\d+\.?\d*)%", content)

          if recommendation_match and probability_match:
              recommendation = recommendation_match.group(1)
              probability = float(probability_match.group(1))
              reports.append({"asset": asset_name, "recommendation": recommendation, "probability": probability})
          else:
              print(f"Could not extract all data from {filename}")
    except FileNotFoundError:
      print(f"File not found: {filename}")

reports_sorted = sorted(reports, key=lambda x: x["probability"])

print("Asset Report Summary (Sorted by Take Profit Probability):")
for report in reports_sorted:
