import os
import re

def analyze_reports(directory="response"):
    reports = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[len("response/"): -len(".md")]  # Extract Asset Name
            filepath = os.path.join(directory, filename)
            try:
                with open(filepath, 'r') as f:
                    content = f.read()

                    # Use regular expressions to find the data.  These are VERY basic examples!
                    recommendation_match = re.search(r"(SHORT|LONG)", content, re.IGNORECASE)
                    probability_match = re.search(r"Probability: (\d+)%", content)

                    recommendation = recommendation_match.group(1).upper() if recommendation_match else "N/A"
                    probability = int(probability_match.group(1)) / 100.0 if probability_match else 0.0

                    reports.append({
                        "asset_name": asset_name,
                        "recommendation": recommendation,
                        "profit_probability": probability
                    })

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")

    # Sort by profit probability (descending)
    sorted_reports = sorted(reports, key=lambda x: x["profit_probability"], reverse=True)

    print("Trading Recommendation Summary (Sorted by Profit Probability):")
    for report in sorted_reports:
        print(f"Asset: {report['asset_name']}, Recommendation: {report['recommendation']}, Probability: {report['profit_probability']:.0%}")


# Example usage (assuming files are in a "response" directory):
