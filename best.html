import os
import re

def parse_report(filepath):
    """
    Parses a single report file and extracts relevant information.
    Returns a dictionary containing:
        'asset_name': str,
        'position': str,
        'entry_price': float,
        'stop_loss': float,
        'profit_probability': float
    Returns None if parsing fails or the report is incomplete
    """
    try:
        with open(filepath, 'r') as f:
            content = f.read()

            # Use regular expressions to extract data. Adjust regex based on the actual report format.
            asset_name_match = re.search(r"response/(.+)\.md", filepath)
            if not asset_name_match:
                return None
            asset_name = asset_name_match.group(1)
            position_match = re.search(r"Recommendation:\s*(SHORT|LONG)", content)
            entry_price_match = re.search(r"Entry Price:\s*([\d.]+)", content)
            stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content)
            probability_match = re.search(r"Profit Probability:\s*([\d.]+)%", content) #or Profit Probability:\s*([\d.]+)

            if not all([position_match, entry_price_match, stop_loss_match, probability_match]):
                return None #incomplete data

            position = position_match.group(1)
            entry_price = float(entry_price_match.group(1))
            stop_loss = float(stop_loss_match.group(1))
            profit_probability = float(probability_match.group(1))

            return {
                'asset_name': asset_name,
                'position': position,
                'entry_price': entry_price,
                'stop_loss': stop_loss,
                'profit_probability': profit_probability
            }
    except Exception as e:
        print(f"Error parsing {filepath}: {e}")
        return None

def main():
    report_data = []
    # Assuming your report files are in the 'response' directory
    for filename in os.listdir("response"):
        if filename.endswith(".md"):
            filepath = os.path.join("response", filename)
            report = parse_report(filepath)
            if report:
                report_data.append(report)

    # Sort by profit probability in descending order
    report_data.sort(key=lambda x: x['profit_probability'], reverse=True)

    # Print the results
    for report in report_data:
        print(f"Asset: {report['asset_name']}, Probability: {report['profit_probability']:.2f}%")

if __name__ == "__main__":
    main()
