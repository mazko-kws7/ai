import os
import re

def extract_report_data(filepath):
    """
    Extracts the asset name and profit probability from a report file.

    Args:
        filepath (str): The path to the report file (e.g., 'response/AAPL.md').

    Returns:
        tuple: A tuple containing the asset name (str) and profit probability (float).
               Returns None if the profit probability cannot be extracted.
    """
    try:
        asset_name = os.path.splitext(os.path.basename(filepath))[0]  # Extract ASSET_NAME from filename
        with open(filepath, 'r') as f:
            content = f.read()

        # Extract profit probability using regular expressions
        match = re.search(r"Profit Probability:\s*([\d\.]+)", content, re.IGNORECASE)  #Case insensitive
        if match:
            profit_probability = float(match.group(1)) # Get matched group and convert to float
            return asset_name, profit_probability
        else:
            print(f"Warning: Could not find Profit Probability in {filepath}")
            return asset_name, None # Or raise an exception, depending on error handling strategy

    except FileNotFoundError:
        print(f"Error: File not found: {filepath}")
        return None
    except Exception as e:
        print(f"Error processing {filepath}: {e}")
        return None


def create_summary(filepaths):
    """
    Creates a sorted summary of the reports based on profit probability.

    Args:
        filepaths (list): A list of filepaths to the report files.

    Returns:
        list: A list of strings, where each string is a summary of a report,
              sorted by profit probability (highest first).
    """

    report_data = []
    for filepath in filepaths:
        data = extract_report_data(filepath)
        if data:
            asset_name, profit_probability = data
            if profit_probability is not None:
                report_data.append({"asset": asset_name, "probability": profit_probability})
            else:
                print(f"Skipping {filepath} due to missing profit probability.")

    # Sort by probability in descending order
    report_data.sort(key=lambda x: x["probability"], reverse=True)

    summaries = []
    for item in report_data:
        summaries.append(f"Asset: {item['asset']}, Profit Probability: {item['probability']:.2f}") # Format to 2 decimal places


    return summaries
