import os
import re

def extract_data_from_file(filename):
    """Extracts trading data from a report file."""
    asset_name = filename.split('/')[-1].replace('.md', '')  # Extract asset name from filename
    try:
        with open(filename, 'r') as f:
            content = f.read()
            # Use regex to extract information.  Adjust regex patterns based on actual file format.
            recommendation_match = re.search(r"Recommendation:\s*(LONG|SHORT)", content)
            entry_match = re.search(r"Entry Price:\s*([\d.]+)", content)
            stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content)
            probability_match = re.search(r"Probability:\s*([\d.]+)%", content)  # Changed to capture the percentage value

            recommendation = recommendation_match.group(1) if recommendation_match else "N/A"
            entry_price = float(entry_match.group(1)) if entry_match else 0.0
            stop_loss = float(stop_loss_match.group(1)) if stop_loss_match else 0.0
            probability = float(probability_match.group(1)) if probability_match else 0.0

        return {
            'asset': asset_name,
            'probability': probability,
            'recommendation': recommendation,
            'entry': entry_price,
            'stop_loss': stop_loss
        }
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return None
    except Exception as e:
        print(f"Error processing {filename}: {e}")
        return None

def main():
    report_files = [f"response/{f}" for f in os.listdir("response") if f.endswith(".md")]  # Get a list of all md files

    reports = []
    for file in report_files:
      data = extract_data_from_file(file)
      if data: #make sure we didn't get None from the extract function
        reports.append(data)

    # Sort reports by probability in descending order
    reports.sort(key=lambda x: x['probability'], reverse=True)

    print("**Report Summary (Sorted by Take Profit Probability)**")
    for report in reports:
        print(f"*   **Asset:** {report['asset']} - **Probability:** {report['probability']}% - **Recommendation:** {report['recommendation']} - **Entry:** {report['entry']} - **Stop Loss:** {report['stop_loss']}")

    • Use ([\d.]+) to capture a number (integer or decimal). The parentheses create a capturing group.
    • The ? at the end of the * makes it lazy matching so it will stop at the correct character.
  • Data Conversion: Converts the extracted entry_price, stop_loss, and probability to float to allow for numerical sorting.
  • List Comprehension for Filenames: report_files = [f"response/{f}" for f in os.listdir("response") if f.endswith(".md")] efficiently creates a list of all .md files in the “response” directory.
