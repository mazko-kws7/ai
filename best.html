import os
import glob
import re

def analyze_reports(directory="response"):
    """Analyzes trading reports in a directory and prints a summary."""
    reports = []
    for filename in glob.glob(os.path.join(directory, "*.md")):
        with open(filename, "r") as f:
            content = f.read()

        # Extract information using regular expressions (adjust patterns as needed)
        asset_name = os.path.splitext(os.path.basename(filename))[0] # Remove the .md extension
        position_match = re.search(r"Position:\s*(\w+)", content)
        probability_match = re.search(r"Profit Probability:\s*(\d+)%", content) # added % to regex
        entry_match = re.search(r"Entry:\s*([\d.]+)", content)
        stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content)

        if position_match and probability_match and entry_match and stop_loss_match:
            position = position_match.group(1).strip()
            probability = int(probability_match.group(1))
            entry = float(entry_match.group(1))
            stop_loss = float(stop_loss_match.group(1))


            reports.append({
                "asset": asset_name,
                "probability": probability,
                "position": position,
                "entry": entry,
                "stop_loss": stop_loss
            })
        else:
            print(f"Warning: Could not parse report {filename}")

    # Sort by probability in descending order
    reports.sort(key=lambda x: x["probability"], reverse=True)

    for report in reports:
        print(f"{report['asset']}: {report['position']}, Profit Probability: {report['probability']}%")

# Example usage:
analyze_reports()
