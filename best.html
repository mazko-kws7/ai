import os
import re

def analyze_reports(report_lines):
    """Analyzes trading reports and returns a sorted summary."""

    report_data = []
    for line in report_lines:
        if line.startswith("==> response/"):
            asset_name = line.split("/")[1].split(".")[0]  # Extract asset name

            try:
                with open(f"response/{asset_name}.md", "r") as f:
                    content = f.read()

                # ** ADJUST THIS REGEX TO MATCH YOUR PROBABILITY FORMAT **
                # Example: Matches "Probability: 75%" or "Probability: 0.75"
                match = re.search(r"Probability: (\d+\.?\d*)%", content) #Example: Probability: 75%
                #match = re.search(r"probability: (\d+\.?\d*)", content, re.IGNORECASE) #Example: probability: 0.75
                if match:
                    probability = float(match.group(1))
                    if "%" not in match.group(0):
                        probability = probability * 100
                    report_data.append({"asset": asset_name, "probability": probability})
                else:
                    print(f"Warning: Could not extract probability from {asset_name}.md")
                    report_data.append({"asset": asset_name, "probability": 0}) #Assign 0 if could not find probability

            except FileNotFoundError:
                print(f"Error: File response/{asset_name}.md not found.")
                report_data.append({"asset": asset_name, "probability": 0}) #Assign 0 if could not find file

    # Sort by probability in descending order
    sorted_reports = sorted(report_data, key=lambda x: x["probability"], reverse=True)
    return sorted_reports

# --- Main Execution ---

#Provide the input lines as a list of strings
report_lines = [
    "==> response/AAPL.md <==",
    "==> response/GOOG.md <==",
    "==> response/MSFT.md <==",
    "==> response/TSLA.md <=="
]

sorted_summaries = analyze_reports(report_lines)

for summary in sorted_summaries:
    print(f"Asset: {summary['asset']}, Profit Probability: {summary['probability']:.0f}%")
