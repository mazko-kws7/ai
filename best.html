import re

def summarize_reports(report_lines, report_content_fetcher):
    """
    Summarizes trading reports, sorts by profit probability, and prints the results.

    Args:
        report_lines: A list of strings, where each string is a line like "==> response/ASSET_NAME.md <==".
        report_content_fetcher: A function that takes an asset name and returns a dictionary
                                 containing the report details (including "profit_probability").  If the
                                 report cannot be fetched, it should return None.
    """

    reports = []
    for line in report_lines:
        match = re.search(r"response/(\w+)\.md", line) #Extract asset name
        if match:
            asset_name = match.group(1)
            report_data = report_content_fetcher(asset_name) #get report data

            if report_data:
              reports.append(report_data)

    # Sort reports by profit_probability in descending order
    reports.sort(key=lambda x: x["profit_probability"], reverse=True)

    # Print the sorted summaries
    for report in reports:
        print(f"{report['asset_name']}: {report['profit_probability'] * 100:.0f}% Profit Probability")
