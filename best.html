import glob
import re

def analyze_reports():
    """Analyzes trading reports and prints a sorted summary."""

    report_data = []

    for filename in glob.glob("response/*.md"):
        try:
            with open(filename, "r") as f:
                content = f.read()

                # Extract profit probability using regular expression
                match = re.search(r"Profit Probability: (\d+)%", content)
                if match:
                    probability = int(match.group(1))
                    asset_name = filename.split("/")[1].split(".")[0]  # Extract asset name from filename
                    report_data.append((asset_name, probability))
                else:
                    print(f"Warning: Could not find Profit Probability in {filename}")

        except FileNotFoundError:
            print(f"Error: File not found: {filename}")
        except Exception as e:
            print(f"Error processing {filename}: {e}")

    # Sort by probability
    report_data.sort(key=lambda item: item[1])

    # Print the sorted results
    for asset_name, probability in report_data:
        print(f"{asset_name}: {probability}%")


#Call the method
analyze_reports()
