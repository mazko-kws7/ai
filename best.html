import os
import re

def analyze_reports(directory="response"):
    """Analyzes trading position reports and sorts them by profit probability."""
    report_data = []

    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename.replace("response/", "").replace(".md", "")  # Extract asset name
            filepath = os.path.join(directory, filename)

            try:
                with open(filepath, "r") as f:
                    content = f.read()
                    # Use a regular expression to find the probability
                    match = re.search(r"Probability:\s*(\d+)%", content) # Example regex
                    if match:
                        probability = int(match.group(1)) # Convert to int

                        report_data.append({"asset": asset_name, "probability": probability})
                    else:
                        print(f"Warning: Could not find probability in {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")

    # Sort by probability in descending order
    sorted_reports = sorted(report_data, key=lambda x: x["probability"], reverse=True)

    # Print results
    print("**Sorted Trading Position Recommendations (by Profit Probability):**\n")
    for report in sorted_reports:
        print(f"*   **Asset:** {report['asset']}, **Probability:** {report['probability']}%")


# Example usage (assuming your reports are in the 'response' directory)
analyze_reports()
