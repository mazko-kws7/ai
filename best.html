import os
import re

def extract_data(filepath):
    """Extracts asset name, probability, and recommendation from a file."""
    asset_name = os.path.splitext(os.path.basename(filepath))[0].split("_")[1] # Get ASSET_NAME from filename
    probability = None
    recommendation = None

    try:
        with open(filepath, 'r') as f:
            content = f.read()
            # Use regex to find the probability and recommendation
            probability_match = re.search(r"Profit Probability: (\d+)%", content)
            recommendation_match = re.search(r"Trading Recommendation: (LONG|SHORT)", content)

            if probability_match:
                probability = int(probability_match.group(1))
            if recommendation_match:
                recommendation = recommendation_match.group(1)

    except FileNotFoundError:
        print(f"Error: File not found: {filepath}")
        return None

    if probability is not None and recommendation is not None:
        return {"asset": asset_name, "probability": probability, "recommendation": recommendation}
    else:
        return None

def generate_summary(directory="response"):
    """Generates a summary of trading recommendations from files in a directory."""
    data = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            filepath = os.path.join(directory, filename)
            extracted_data = extract_data(filepath)
            if extracted_data:
                data.append(extracted_data)

    # Sort by probability in descending order
    data.sort(key=lambda x: x["probability"], reverse=True)

    print("Summary of Trading Recommendations (Sorted by Profit Probability):")
    for item in data:
        print(f"*   **Asset:** {item['asset']}, **Probability:** {item['probability']}%, **Recommendation:** {item['recommendation']}")
