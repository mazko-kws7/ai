import os
import re

def extract_report_data(filename):
  """Extracts asset name and profit probability from a report file."""
  try:
    with open(filename, 'r') as f:
      content = f.read()

      asset_match = re.search(r"Asset:\s*(\w+)", content)
      probability_match = re.search(r"Profit Probability:\s*(\d+)%", content)

      if asset_match and probability_match:
        asset_name = asset_match.group(1)
        profit_probability = int(probability_match.group(1))
        return asset_name, profit_probability
      else:
        print(f"Warning: Could not extract data from {filename}")
        return None, None

  except FileNotFoundError:
    print(f"Error: File not found: {filename}")
    return None, None

def generate_sorted_summary(report_files):
  """Generates and sorts the summary of reports."""
  report_data = []
  for filename in report_files:
    asset_name, profit_probability = extract_report_data(filename)
    if asset_name and profit_probability:
      report_data.append((asset_name, profit_probability))

  report_data.sort(key=lambda item: item[1], reverse=True)  # Sort by probability

  for asset_name, profit_probability in report_data:
    print(f"{asset_name}: Profit Probability: {profit_probability}%")


# Example Usage (assuming your files are in the 'response' directory):
report_files = [f"response/{f}" for f in os.listdir("response") if f.endswith(".md")]

if report_files:
  generate_sorted_summary(report_files)
else:
  print("No report files found in the 'response' directory.")
