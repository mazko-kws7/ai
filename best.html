    import re

def extract_data(filepath):
    asset_name = filepath.split('/')[1].split('.')[0]
    try:
        with open(filepath, 'r') as f:
            content = f.read()

            # Use regular expressions to find the data (adjust regex patterns as needed)
            position_match = re.search(r"Position:\s*(SHORT|LONG)", content, re.IGNORECASE)
            probability_match = re.search(r"Profit Probability:\s*(\d+)%", content, re.IGNORECASE)

            position = position_match.group(1) if position_match else "UNKNOWN"
            probability = int(probability_match.group(1)) if probability_match else 0

            return {"asset_name": asset_name, "position": position, "probability": probability}

    except FileNotFoundError:
        print(f"Error: File not found: {filepath}")
        return {"asset_name": asset_name, "position": "ERROR", "probability": 0}
    except Exception as e:
        print(f"Error processing {filepath}: {e}")
        return {"asset_name": asset_name, "position": "ERROR", "probability": 0}
