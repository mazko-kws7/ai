def analyze_reports(report_list):
    """
    Analyzes a list of trading reports and returns a sorted summary.

    Args:
        report_list: A list of dictionaries, where each dictionary represents a report.
                     Each dictionary should have keys: "asset_name", "profit_probability".

    Returns:
        A list of strings, sorted by profit probability in descending order,
        containing the asset name and profit probability.
    """
    summaries = []
    for report in report_list:
        asset_name = report["asset_name"]
        profit_probability = report["profit_probability"]
        summaries.append(f"{asset_name}: {profit_probability:.2f}")  # Format probability to 2 decimal places

    # Sort by profit probability in descending order
    summaries.sort(key=lambda x: float(x.split(":")[1]), reverse=True)  # Extract probability to sort
    return summaries

def read_report_file(filename):
    """Reads a report file and extracts the relevant information."""
    try:
        with open(filename, "r") as f:
            content = f.read()

            # Use regular expressions to extract the information
            asset_name_match = re.search(r"response/(.*).md", filename)
            recommendation_match = re.search(r"Recommendation: (SHORT|LONG)", content)
            entry_price_match = re.search(r"Entry Price: ([\d.]+)", content)
            stop_loss_match = re.search(r"Stop Loss: ([\d.]+)", content)
            profit_probability_match = re.search(r"Profit Probability: ([\d.]+)", content)

            if asset_name_match and recommendation_match and entry_price_match and stop_loss_match and profit_probability_match:
                asset_name = asset_name_match.group(1)
                recommendation = recommendation_match.group(1)
                entry_price = float(entry_price_match.group(1))
                stop_loss = float(stop_loss_match.group(1))
                profit_probability = float(profit_probability_match.group(1))

                return {
                    "asset_name": asset_name,
                    "recommendation": recommendation,
                    "entry_price": entry_price,
                    "stop_loss": stop_loss,
                    "profit_probability": profit_probability,
                }
            else:
                print(f"Could not parse report file: {filename}")
                return None
    except FileNotFoundError:
        print(f"File not found: {filename}")
        return None
