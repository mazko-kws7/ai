Example (Python):

import os
import re

def analyze_reports(directory="response"):
    """Analyzes trading reports, extracts profit probabilities, and sorts them."""
    reports = []

    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[:-3]  # Remove ".md"
            filepath = os.path.join(directory, filename)
            try:
                with open(filepath, "r") as f:
                    content = f.read()
                    match = re.search(r"Profit Probability:\s*(\d+)%", content)
                    if match:
                        probability = int(match.group(1))
                        reports.append({"asset": asset_name, "probability": probability})
                    else:
                        print(f"Warning: Could not find probability in {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")


    sorted_reports = sorted(reports, key=lambda x: x["probability"], reverse=True)

    for report in sorted_reports:
        print(f"{report['asset']}: {report['probability']}% Profit Probability")

# Example usage:
analyze_reports()
