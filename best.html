import re
import os

def analyze_reports(report_files):
    results = []
    for file_path in report_files:
        try:
            with open(file_path, 'r') as f:
                content = f.read()
            #Use filename, for instance file_path = 'response/AAPL.md', then asset_name = 'AAPL'
            asset_name = os.path.splitext(os.path.basename(file_path))[0]
            match = re.search(r"Profit Probability: (\d+)%", content) #Find the probability line
            if match:
                probability = int(match.group(1)) #Extract only the digit part
                results.append({"asset": asset_name, "probability": probability})
            else:
                print(f"Warning: Could not find profit probability in {file_path}")

        except FileNotFoundError:
            print(f"Error: File not found: {file_path}")
        except Exception as e:
            print(f"Error processing {file_path}: {e}") #In case other issues come up

    # Sort by probability (descending)
    sorted_results = sorted(results, key=lambda x: x["probability"], reverse=True)

    # Print the summary
    for item in sorted_results:
        print(f"{item['asset']}: {item['probability']}%")

# Example usage (replace with your actual list of files)
report_files = ["response/GOLD.md", "response/BTC.md", "response/EURUSD.md", "response/AAPL.md", "response/OIL.md"]
analyze_reports(report_files)
