import os

def analyze_reports(directory="response"):
    """
    Analyzes trading reports in a directory, extracts relevant information,
    and sorts them by profit probability.

    Args:
        directory (str): The directory containing the report files.  Defaults to "response".

    Returns:
        list: A list of tuples, where each tuple contains (asset_name, recommendation, probability).
              Returns an empty list if the directory doesn't exist or an error occurs.
    """

    try:
        report_data = []
        for filename in os.listdir(directory):
            if filename.endswith(".md"):
                asset_name = filename[:-3]  # Remove ".md" extension
                filepath = os.path.join(directory, filename)

                with open(filepath, 'r') as f:
                    content = f.read()

                    # Simple parsing (adjust based on your report format)
                    recommendation = None
                    probability = None
                    for line in content.splitlines():
                        if "Recommendation:" in line:
                            recommendation = line.split(":")[1].strip()
                        if "Probability:" in line:
                            try:
                                probability = float(line.split(":")[1].strip().replace("%",""))
                            except ValueError:
                                print(f"Warning: Could not parse probability in {filename}. Skipping.")
                                probability = None
                                break # Skip to next file, as crucial data is missing

                    if recommendation and probability is not None:  # Ensure both values were found
                        report_data.append((asset_name, recommendation, probability))
                    else:
                        print(f"Warning: Could not extract all info from {filename}. Skipping.")


        # Sort by probability (descending)
        sorted_reports = sorted(report_data, key=lambda x: x[2], reverse=True)

        return sorted_reports

    except FileNotFoundError:
        print(f"Error: Directory '{directory}' not found.")
        return []
    except Exception as e:
        print(f"An error occurred: {e}")
        return []


def print_report_summary(sorted_reports):
    """
    Prints a formatted summary of the trading reports.

    Args:
        sorted_reports (list): A list of tuples containing (asset_name, recommendation, probability),
                                 sorted by probability in descending order.
    """

    if not sorted_reports:
        print("No reports to summarize.")
        return

    print("Summary of Trading Reports (Sorted by Profit Probability):")
    for asset_name, recommendation, probability in sorted_reports:
        print(f"{asset_name}: Recommendation: {recommendation}, Probability: {probability:.2f}%")
