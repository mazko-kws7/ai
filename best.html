import re

def summarize_reports(filenames):
    reports = []
    for filename in filenames:
        try:
            with open(filename, 'r') as f:
                content = f.read()

            # Extract asset name from filename
            asset = filename.split('/')[-1].replace('.md', '')

            # Use regex to extract the data, adjust as needed
            recommendation_match = re.search(r"Recommendation:\s*(LONG|SHORT)", content)
            probability_match = re.search(r"Profit Probability:\s*(\d+\.?\d*)%", content)
            entry_match = re.search(r"Entry Price:\s*(\d+\.?\d*)", content)
            stop_loss_match = re.search(r"Stop Loss:\s*(\d+\.?\d*)", content)

            if recommendation_match and probability_match:
                recommendation = recommendation_match.group(1)
                probability = float(probability_match.group(1)) / 100.0
                entry = float(entry_match.group(1)) if entry_match else None
                stop_loss = float(stop_loss_match.group(1)) if stop_loss_match else None

                reports.append({
                    'asset': asset,
                    'recommendation': recommendation,
                    'entry': entry,
                    'stop_loss': stop_loss,
                    'probability': probability
                })
            else:
                print(f"Warning: Could not extract data from {filename}")

        except FileNotFoundError:
            print(f"Error: File not found: {filename}")
        except Exception as e:
            print(f"Error processing {filename}: {e}")

    reports.sort(key=lambda x: x['probability'], reverse=True)

    output = ""
    for report in reports:
        output += f"{report['asset']}: Profit Probability = {report['probability']:.2f}\n"

    return output

# Example usage (assuming you have a list of filenames)
filenames = ["response/AAPL.md", "response/GOOG.md", "response/TSLA.md"] # Replace with your actual list of filenames
summary = summarize_reports(filenames)
print(summary)
