```python
import os
import re

report_data = []

# Assuming reports are in the 'response' directory
report_dir = "response"

# Iterate through all files in the directory
for filename in os.listdir(report_dir):
    if filename.endswith(".md"):
        filepath = os.path.join(report_dir, filename)
        asset_name = filename[:-3]  # Remove ".md" extension

        with open(filepath, "r") as f:
            content = f.read()

            # Use regular expressions to extract the required values
            recommendation_match = re.search(r"Recommendation:\s*(\w+)", content)
            probability_match = re.search(r"Probability of Take Profit:\s*([\d.]+)", content)

            if recommendation_match and probability_match:
                recommendation = recommendation_match.group(1)
                probability = float(probability_match.group(1))

                report_data.append({
                    "asset": asset_name,
                    "recommendation": recommendation,
                    "probability": probability
                })
            else:
                print(f"Warning: Could not extract data from {filename}")

# Sort the data by probability in descending order
report_data.sort(key=lambda x: x["probability"], reverse=True)

# Generate the output string
output = "Here's a summary of the reports, sorted by probability of take profit (highest to lowest):\n\n"
for item in report_data:
    output += f"*   **{item['asset']}**: Recommendation: {item['recommendation']}, Probability: {item['probability']:.2f}\n"

print(output)
