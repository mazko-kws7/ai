import os
import re

def extract_data(filename):
    """Extracts the trading recommendation, entry price, stop loss, and profit probability from a report file.
    Returns a dictionary with the extracted information."""
    try:
        with open(filename, 'r') as f:
            content = f.read()

            # Use regex to find the trading recommendation (LONG or SHORT)
            recommendation_match = re.search(r"(LONG|SHORT)", content)
            recommendation = recommendation_match.group(1) if recommendation_match else "UNKNOWN"

            # Use regex to find the entry price
            entry_price_match = re.search(r"Entry Price:\s*([\d.]+)", content, re.IGNORECASE)
            entry_price = float(entry_price_match.group(1)) if entry_price_match else None

            # Use regex to find the stop loss
            stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)", content, re.IGNORECASE)
            stop_loss = float(stop_loss_match.group(1)) if stop_loss_match else None

            # Use regex to find the profit probability
            probability_match = re.search(r"Profit Probability:\s*([\d.]+)%", content, re.IGNORECASE)
            probability = float(probability_match.group(1)) if probability_match else None

            return {
                "recommendation": recommendation,
                "entry_price": entry_price,
                "stop_loss": stop_loss,
                "probability": probability
            }
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return None
    except Exception as e:
        print(f"Error processing file {filename}: {e}")
        return None


def process_reports(directory="response"):
    """Processes all report files in the given directory, extracts data, and sorts by profit probability."""
    report_data = []

    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            filepath = os.path.join(directory, filename)
            asset_name = filename.replace("response/", "").replace(".md", "")  # Extract the asset name
            data = extract_data(filepath)

            if data and data["probability"] is not None:
                report_data.append({
                    "asset": asset_name,
                    "recommendation": data["recommendation"],
                    "entry_price": data["entry_price"],
                    "stop_loss": data["stop_loss"],
                    "probability": data["probability"]
                })

    # Sort by profit probability (descending)
    sorted_reports = sorted(report_data, key=lambda x: x["probability"], reverse=True)

    return sorted_reports


def generate_summary(sorted_reports):
    """Generates a summary string from the sorted report data."""
    summary = ""
    for report in sorted_reports:
        summary += f"Asset: {report['asset']}, Recommendation: {report['recommendation']}, Probability: {report['probability']}%\n"
    return summary
