import re

def process_reports(reports):
    """Processes a list of trading reports, extracts key information,
    and sorts them by profit probability.

    Args:
        reports: A list of tuples, where each tuple contains:
            - filename: The name of the report file (e.g., "response/AAPL.md")
            - content: The content of the report as a string.

    Returns:
        A list of strings, where each string is a summary of a report,
        sorted by profit probability in descending order.
    """

    report_summaries = []
    for filename, content in reports:
        # Extract asset name from filename (assuming format response/ASSET_NAME.md)
        asset_name = filename.split('/')[1].split('.')[0]

        # Use regex to extract the information
        # Updated for more robust extraction and handling of missing values
        match = re.search(r"POSITION:\s*(\w+)\s*"
                          r"ENTRY:\s*([0-9.]+)\s*"
                          r"STOP LOSS:\s*([0-9.]+)\s*"
                          r"PROBABILITY:\s*([0-9.]+)", content, re.IGNORECASE)


        if match:
            position = match.group(1).upper()
            entry_price = float(match.group(2))
            stop_loss = float(match.group(3))
            probability = float(match.group(4))

            summary = f"{asset_name}: {position} - Probability: {probability:.2f}"
            report_summaries.append((summary, probability))  # Store probability for sorting
        else:
            print(f"Warning: Could not extract data from {filename}")
            # Handle the case where the expected data is not found.
            # This could be adding a default summary or skipping the report.
            summary = f"{asset_name}: Data extraction failed"
            report_summaries.append((summary, 0.0)) # default zero probability

    # Sort by probability in descending order
    sorted_summaries = [summary for summary, probability in sorted(report_summaries, key=lambda item: item[1], reverse=True)]
    return sorted_summaries
