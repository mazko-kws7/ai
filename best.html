import os
import re  # for regular expressions to extract data
import glob #to find all report files

def extract_report_data(filepath):
    """Extracts asset name, position, and profit probability from a report file.
    Returns a dictionary or None if extraction fails.
    """
    asset_name = os.path.splitext(os.path.basename(filepath))[0] #extract asset name
    try:
        with open(filepath, 'r') as f:
            content = f.read()

            # Use regular expressions to extract the data.  Adjust these patterns
            # to match the EXACT format of your report files.
            position_match = re.search(r"Position:\s*(LONG|SHORT)", content)
            probability_match = re.search(r"Profit Probability:\s*(\d+)%", content)
            #entry_price_match = re.search(r"Entry Price:\s*([\d.]+)",content) #extract entry price
            #stop_loss_match = re.search(r"Stop Loss:\s*([\d.]+)",content) #extract stop loss

            if position_match and probability_match:
                position = position_match.group(1)
                probability = int(probability_match.group(1))
                #entry_price = float(entry_price_match.group(1))
                #stop_loss = float(stop_loss_match.group(1))

                return {
                    'asset_name': asset_name,
                    'position': position,
                    'probability': probability,
                    #'entry_price': entry_price,
                    #'stop_loss': stop_loss
                }
            else:
                print(f"Warning: Could not extract data from {filepath}")
                return None
    except FileNotFoundError:
        print(f"Error: File not found: {filepath}")
        return None
    except Exception as e:
        print(f"Error processing {filepath}: {e}")
        return None


def summarize_reports(report_dir="response"): #directory containing the reports
    """Summarizes the reports in the given directory."""

    report_files = glob.glob(os.path.join(report_dir, "*.md")) #list of report files

    report_data = []
    for filepath in report_files:
        data = extract_report_data(filepath)
        if data:
            report_data.append(data)

    # Sort by profit probability (descending)
    sorted_reports = sorted(report_data, key=lambda x: x['probability'], reverse=True)

    # Print the summary
    print("Report Summary (Sorted by Profit Probability):")
    for report in sorted_reports:
        print(f"  Asset: {report['asset_name']}, Position: {report['position']}, Probability: {report['probability']}%")


# Run the summarization (replace "response" with your actual directory)
summarize_reports()
