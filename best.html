import os
import re

def analyze_reports(directory="response"):  # Changed to a function for clarity
    """
    Analyzes trading reports in a directory, extracts asset name,
    take profit probability, and sorts by probability.

    Args:
        directory (str): The directory containing the reports.

    Returns:
        list: A sorted list of dictionaries, where each dictionary contains
              'asset', 'probability'.  Returns an empty list if no reports
              are found.
    """

    reports = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[:-3]  # Remove ".md"
            filepath = os.path.join(directory, filename)

            try:
                with open(filepath, 'r') as f:
                    content = f.read()

                # Use regular expression to find the probability
                match = re.search(r"Take Profit Probability:\s*([0-9.]+)", content)
                if match:
                    probability = float(match.group(1))
                    reports.append({"asset": asset_name, "probability": probability})
                else:
                    print(f"Warning: Could not extract probability from {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")

    # Sort by probability (descending)
    reports.sort(key=lambda x: x["probability"], reverse=True)
    return reports


def print_summary(reports):
    """Prints a formatted summary of the reports."""
    if not reports:
        print("No reports found or processed.")
        return

    print("Trading Report Summary (Sorted by Take Profit Probability):")
    for report in reports:
        print(f"Asset: {report['asset']}, Probability: {report['probability']:.2f}")
