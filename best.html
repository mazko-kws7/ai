import glob
import re

def summarize_reports(report_pattern="response/*.md"):
    """
    Analyzes trading reports, extracts key information, and sorts by profit probability.
    """
    reports = []

    for filename in glob.glob(report_pattern):  # Find all report files
        try:
            asset_name = filename.split("/")[1].split(".")[0]  # Extract asset name

            with open(filename, "r") as f:
                content = f.read()

            # Use regex to extract information - adjust patterns based on your report format!
            position_match = re.search(r"Recommendation:\s*(SHORT|LONG)", content)
            probability_match = re.search(r"Profit Probability:\s*([\d.]+)%", content) # Adjusted to match % symbol

            if position_match and probability_match:
                position = position_match.group(1)
                probability = float(probability_match.group(1))

                reports.append({
                    "asset": asset_name,
                    "position": position,
                    "probability": probability
                })
            else:
                print(f"Warning: Could not extract data from {filename}")

        except FileNotFoundError:
            print(f"Error: File not found: {filename}")
        except Exception as e:
            print(f"Error processing {filename}: {e}")

    # Sort by profit probability (descending)
    reports.sort(key=lambda x: x["probability"], reverse=True)

    # Print the sorted summary
    for report in reports:
        print(f"Asset: {report['asset']}, Recommendation: {report['position']}, Profit Probability: {report['probability']:.1f}%")


# Example usage (if you had the files):
summarize_reports()
