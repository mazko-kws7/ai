import os
import re

def analyze_reports(directory="response"):
    report_data = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename.replace("response/", "").replace(".md", "") # remove response/ and .md
            try:
                with open(os.path.join(directory, filename), "r") as f:
                    content = f.read()
                    #EXAMPLE parsing using Regex to find Profit probability
                    # Adjust REGEX to your file contents and find all parameters
                    match = re.search(r"Profit Probability:\s*([0-9.]+)", content)
                    if match:
                        probability = float(match.group(1))
                        report_data.append({"asset": asset_name, "probability": probability})
                    else:
                        print(f"Warning: Could not find probability in {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filename}")

    # Sort by probability (descending)
    sorted_reports = sorted(report_data, key=lambda x: x["probability"], reverse=True)

    # Print results
    for report in sorted_reports:
        print(f"*   Asset: {report['asset']}, Probability: {report['probability']:.2f}")
