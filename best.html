import re

def analyze_reports(report_data):
    """
    Analyzes trading reports, extracts information, and sorts by profit probability.

    Args:
      report_data: A string containing all the report data, separated by lines,
                   including the '==> response/ASSET_NAME.md <==' lines and the
                   report contents.

    Returns:
      A list of strings, each a summary of a report, sorted by profit probability
      (highest first).  Returns an empty list if there are issues.
    """
    reports = report_data.split("==> ") #Splitting based on the start of each report.
    report_summaries = []

    for report in reports:
        if not report.strip():  #Skip empty strings
            continue

        try:
            # Extract asset name (using regex)
            match = re.search(r"response/(?P<asset_name>[A-Za-z0-9]+)\.md <==\n(?P<content>.*)", report, re.DOTALL)
            if not match:
                print(f"Warning: Could not extract asset name from report: {report[:50]}...") #Print first 50 characters
                continue

            asset_name = match.group("asset_name")
            content = match.group("content")


            # Extract profit probability (using regex, assumes consistent format)
            probability_match = re.search(r"Profit Probability:\s*([0-9.]+)", content)
            if not probability_match:
                print(f"Warning: Could not extract profit probability from report for {asset_name}")
                continue

            probability = float(probability_match.group(1))


            report_summaries.append({"asset": asset_name, "probability": probability})

        except Exception as e:
            print(f"Error processing report: {e}")
            continue  # Skip to the next report
    # Sort the summaries by probability (descending)
    sorted_summaries = sorted(report_summaries, key=lambda x: x["probability"], reverse=True)

    # Format the output
    output = [f"{summary['asset']}: Profit Probability = {summary['probability']}" for summary in sorted_summaries]

    return output
