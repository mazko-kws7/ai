import os
import re  # For regular expressions

def summarize_reports(directory="response"):
    """Summarizes trading reports, sorted by probability."""

    reports = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[:-3]  # Remove ".md"
            filepath = os.path.join(directory, filename)

            try:
                with open(filepath, "r") as f:
                    content = f.read()

                    #  **IMPORTANT:** This is a placeholder.  Adjust the regex
                    # to accurately extract the probability from your file content.
                    match = re.search(r"Probability: (\d+)%", content)
                    if match:
                        probability = int(match.group(1))
                        reports.append((asset_name, probability))
                    else:
                        print(f"Warning: Could not find probability in {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")

    # Sort by probability (descending)
    reports.sort(key=lambda item: item[1], reverse=True)

    print("**Trading Recommendation Summary (Sorted by Take Profit Probability)**")
    for asset_name, probability in reports:
        print(f"*   {asset_name}: Probability: {probability}%")
