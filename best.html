import os
import re

def extract_data(filename):
    """Extracts asset name and probability from a report file."""
    asset_name = filename.split('/')[1].split('.')[0] # Extract ASSET_NAME from filename
    try:
        with open(filename, 'r') as f:
            content = f.read()
            match = re.search(r"Take Profit Probability: (\d\.\d+)", content)
            if match:
                probability = float(match.group(1))
                return asset_name, probability
            else:
                print(f"Warning: No probability found in {filename}")
                return asset_name, None  # Or some default value
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return asset_name, None

def main():
    reports = []
    # Assuming the reports are in the 'response' directory
    for filename in os.listdir('response'):
      if filename.endswith('.md'):
        asset_name, probability = extract_data(os.path.join('response', filename))
        if probability is not None:
          reports.append((asset_name, probability))

    # Sort by probability
    reports.sort(key=lambda item: item[1])  # Sort by the probability (second element)

    # Print the results
    for asset_name, probability in reports:
        print(f"{asset_name}: Take Profit Probability - {probability:.2f}")

if __name__ == "__main__":
