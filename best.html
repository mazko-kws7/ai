import os
import re

def analyze_reports(directory="response"):
    """Analyzes trading reports in a directory and returns a sorted summary."""

    reports = []
    for filename in os.listdir(directory):
        if filename.endswith(".md"):
            asset_name = filename[:-3]  # Remove ".md"
            filepath = os.path.join(directory, filename)

            try:
                with open(filepath, "r") as f:
                    content = f.read()

                    # Use regular expressions to find the information
                    position_match = re.search(r"Position:\s*(LONG|SHORT)", content, re.IGNORECASE)
                    probability_match = re.search(r"Probability:\s*(\d+)%", content, re.IGNORECASE)


                    if position_match and probability_match:
                        position = position_match.group(1).upper()
                        probability = int(probability_match.group(1))

                        reports.append({
                            "asset_name": asset_name,
                            "position": position,
                            "probability": probability
                        })
                    else:
                        print(f"Warning: Could not parse {filename}")

            except FileNotFoundError:
                print(f"Error: File not found: {filepath}")
            except Exception as e:
                print(f"Error processing {filename}: {e}")

    # Sort by probability (descending)
    reports.sort(key=lambda x: x["probability"], reverse=True)

    # Generate output
    print("Trading Position Recommendations (Sorted by Profit Probability):")
    for report in reports:
        print(f"Asset: {report['asset_name']}, Position: {report['position']}, Probability: {report['probability']}%")
