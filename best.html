import re

def summarize_reports(reports_text):
    """
    Analyzes trading reports, extracts key information, and sorts them by
    take profit probability.

    Args:
        reports_text: A string containing all the reports, separated by
                      '==> response/ASSET_NAME.md <=='.  Each report
                      should contain 'LONG' or 'SHORT', 'entry price', and
                      'profit probability'.

    Returns:
        A string containing a sorted summary of the reports, or an error
        message if the input is invalid.
    """

    reports = reports_text.split("==> ")
    summaries = []

    for report in reports:
        if not report.strip():  # Skip empty strings
            continue

        match = re.match(r"response/(?P<asset_name>[^.]+)\.md <==\n(?P<content>.*)", report, re.DOTALL)
        if not match:
            continue
            #return f"Error: Invalid report format: {report[:50]}..."

        asset_name = match.group("asset_name")
        content = match.group("content")

        # Extract information using regular expressions (adjust patterns as needed)
        recommendation_match = re.search(r"(LONG|SHORT)", content)
        probability_match = re.search(r"profit probability:\s*([\d.]+)", content, re.IGNORECASE)
        entry_match = re.search(r"entry price:\s*([\d.]+)", content, re.IGNORECASE)


        if not (recommendation_match and probability_match and entry_match):
            continue  # Skip reports where essential information is missing
            #return f"Error: Could not extract information from report for {asset_name}"

        recommendation = recommendation_match.group(1)
        probability = float(probability_match.group(1))
        entry_price = float(entry_match.group(1))



        summaries.append({
            "asset_name": asset_name,
            "probability": probability,
            "recommendation": recommendation,
            "entry": entry_price
        })

    # Sort by probability (descending)
    summaries.sort(key=lambda x: x["probability"], reverse=True)

    output = ""
    for summary in summaries:
        output += (
            f"Asset: {summary['asset_name']}, Recommendation: {summary['recommendation']}, "
            f"Entry: {summary['entry']}, Probability: {summary['probability']:.2f}\n"
        )

    return output
