import os
import re

report_data = []

for filename in os.listdir("response"):  # Assuming reports are in 'response' directory
    if filename.endswith(".md"):
        asset_name = filename[:-3]  # Remove ".md"
        filepath = os.path.join("response", filename)

        with open(filepath, "r") as f:
            content = f.read()

            # Use regular expressions to extract information
            recommendation_match = re.search(r"Recommendation: (SHORT|LONG)", content)
            probability_match = re.search(r"Profit Probability: (\d+)%", content)

            if recommendation_match and probability_match:
                recommendation = recommendation_match.group(1)
                probability = int(probability_match.group(1))

                report_data.append({
                    "asset_name": asset_name,
                    "recommendation": recommendation,
                    "probability": probability
                })

# Sort by probability
report_data.sort(key=lambda x: x["probability"])

print("Summary of Trading Reports (Sorted by Take Profit Probability):")
for item in report_data:
    print(f"*   **response/{item['asset_name']}.md:** Recommendation: {item['recommendation']}, Probability: {item['probability']}%")
