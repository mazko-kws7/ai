import os
import re

def extract_data(filepath):
    asset_name = os.path.splitext(os.path.basename(filepath))[0]  # Extract ASSET_NAME from filename
    data = {}
    data['asset_name'] = asset_name
    try:
        with open(filepath, 'r') as f:
            content = f.read()
            # Use regular expressions to extract the values.  Adjust these regexes
            # based on the *exact* format of your report files.
            match = re.search(r"Profit Probability:\s*([0-9.]+)", content)
            if match:
                data['profit_probability'] = float(match.group(1))
            else:
                data['profit_probability'] = None # Handle missing probability

            match = re.search(r"Position Recommendation:\s*(LONG|SHORT)", content)
            if match:
                 data['position'] = match.group(1)
            else:
                 data['position'] = None


    except FileNotFoundError:
        print(f"Error: File not found: {filepath}")
        return None

    return data


def main():
    reports = []
    for filename in os.listdir("response"):
        if filename.endswith(".md"):
            filepath = os.path.join("response", filename)
            report_data = extract_data(filepath)
            if report_data: # only add reports for which extraction was successful
                reports.append(report_data)

    # Sort by profit probability
    reports.sort(key=lambda x: x['profit_probability'] if x['profit_probability'] is not None else -1) #Put missing at the beginning

    # Print the results
    for report in reports:
        print(f"{report['asset_name']}: Profit Probability: {report['profit_probability']}")

if __name__ == "__main__":
    main()
