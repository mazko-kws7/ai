def extract_asset_name(filename):
  """Extracts the asset name from the filename."""
  return filename.split("/")[1].split(".")[0]

def parse_report(filename, report_content):
    """Parses the report content and extracts relevant information."""
    asset_name = extract_asset_name(filename)

    try:
        profit_probability = float(re.search(r"Profit Probability:\s*([0-9.]+)", report_content).group(1))
    except (AttributeError, ValueError):
        profit_probability = None
        print(f"Warning: Could not parse Profit Probability in {filename}")

    return {
        "asset_name": asset_name,
        "profit_probability": profit_probability
    }

# Process each report and create summaries
summaries = []
for filename, content in simulated_reports.items():
    data = parse_report(filename, content)
    if data["profit_probability"] is not None:
        summaries.append(data)

# Sort the summaries by profit probability in descending order
summaries_sorted = sorted(summaries, key=lambda x: x["profit_probability"], reverse=True)

# Print the sorted summaries
print("Asset Recommendations Sorted by Profit Probability:")
for summary in summaries_sorted:
    print(f"  {summary['asset_name']}: Profit Probability = {summary['profit_probability']:.2f}")
