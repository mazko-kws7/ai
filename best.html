import re

def analyze_reports(report_data):
    """
    Analyzes trading reports, extracts relevant information, and sorts them by profit probability.

    Args:
        report_data: A string containing the concatenated report data, with reports separated by lines like '==> response/ASSET_NAME.md <=='.

    Returns:
        A list of strings, each summarizing a report, sorted by profit probability in descending order.
    """

    reports = report_data.split("==> response/") #Split into reports
    reports = [report for report in reports if report.strip() != ""] #Clean empty reports
    report_summaries = []

    for report in reports:
        asset_name = re.search(r"(.*)\.md", report).group(1) #Get asset_name from the file path.
        profit_probability_match = re.search(r"Profit Probability: (\d+)%", report)

        if profit_probability_match:
            profit_probability = int(profit_probability_match.group(1))
            report_summaries.append({"asset": asset_name, "probability": profit_probability})
    report_summaries = sorted(report_summaries, key=lambda x: x["probability"], reverse=True)

    output = []
    for report in report_summaries:
        output.append(f"*   **{report['asset']}:** Profit Probability: {report['probability']}%")

    return output
