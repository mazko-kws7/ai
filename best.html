def analyze_reports(report_names):

    and sorts them by profit probability.

        asset_name = report_name.split('/')[1].split('.')[0]  # Extract asset name

        # Extract profit probability using regular expressions
        probability_match = re.search(r"Profit Probability:\s*([0-9.]+%?)", file_content, re.IGNORECASE)
        probability = None
        if probability_match:
            probability_str = probability_match.group(1)
            if "%" in probability_str:
                probability = float(probability_str.replace("%", "")) / 100.0
            else:
                probability = float(probability_str)
        else:
            probability = "N/A"

        report_data.append({

            "profit_probability": probability

    # Sort by profit probability (descending)
    report_data.sort(key=lambda x: x["profit_probability"] if isinstance(x["profit_probability"], float) else -1, reverse=True)

    summary = ""
    for report in report_data:
