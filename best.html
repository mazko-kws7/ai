import glob
import re

def extract_data(filename):
    """Extracts recommendation and probability from a file."""
    try:
        with open(filename, 'r') as f:
            content = f.read()

        recommendation_match = re.search(r"Recommendation:\s*(LONG|SHORT)", content)
        probability_match = re.search(r"Profit Probability:\s*(\d+)%", content)

        if recommendation_match and probability_match:
            recommendation = recommendation_match.group(1)
            probability = int(probability_match.group(1))
            return recommendation, probability
        else:
            return None, None  # Indicate failure to extract
    except FileNotFoundError:
        print(f"Error: File not found: {filename}")
        return None, None
    except Exception as e:
        print(f"Error reading file {filename}: {e}")
        return None, None

def main():
    data = []
    for filename in glob.glob("response/*.md"):  # Find all files matching the pattern
        asset_name = filename.split('/')[1].split('.')[0]  # Extract asset name from filename
        recommendation, probability = extract_data(filename)

        if recommendation and probability:
            data.append({"asset": asset_name, "recommendation": recommendation, "probability": probability})

    # Sort by probability (descending)
    data.sort(key=lambda x: x["probability"], reverse=True)

    print("Summary of Trading Recommendations (Sorted by Profit Probability):")
    for item in data:
        print(f"Asset: {item['asset']}, Recommendation: {item['recommendation']}, Probability: {item['probability']}%")

if __name__ == "__main__":
    main()
