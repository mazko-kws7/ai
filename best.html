import re

def analyze_reports(report_data):
    """
    Analyzes trading reports and extracts asset name, recommendation, and profit probability.

    Args:
      report_data: A list of strings, where each string is a report.
                     Assumes the format: "==> response/ASSET_NAME.md <==\n[Report Content]"

    Returns:
      A list of dictionaries, each dictionary representing a report and sorted by profit probability.
    """

    reports = []
    for report_text in report_data:
        # Split the report into header and content.
        try:
            header, content = report_text.split("\n", 1)  # Split on the first newline
        except ValueError:
            print(f"Skipping report due to unexpected format: {report_text}")
            continue # Skip to the next report if splitting fails.

        # Extract Asset Name
        match = re.search(r"==> response/(.+?)\.md <==", header)
        if not match:
            print(f"Could not extract asset name from header: {header}")
            continue # Skip to next if no name.
        asset_name = match.group(1)

        # Extract recommendation, entry, stop loss, and profit probability.  This assumes simple keyword matching
        # and *requires* consistent formatting within your reports.  Adjust regex as needed.

        recommendation_match = re.search(r"Recommendation:\s*(SHORT|LONG)", content, re.IGNORECASE)
        probability_match = re.search(r"Profit Probability:\s*([\d\.]+)", content, re.IGNORECASE)


        recommendation = recommendation_match.group(1).upper() if recommendation_match else "N/A"
        probability = float(probability_match.group(1)) if probability_match else -1 # use -1 for error/missing

        if probability == -1:
            print(f"Warning: Could not extract all info for {asset_name}")
            continue


        reports.append({
            "asset_name": asset_name,
            "recommendation": recommendation,
            "probability": probability
        })


    # Sort reports by profit probability (ascending)
    sorted_reports = sorted(reports, key=lambda x: x["probability"])

    return sorted_reports
