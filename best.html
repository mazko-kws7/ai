import re

def analyze_reports(report_data):
    reports = []
    current_report = None

    for line in report_data.splitlines():
        if line.startswith("==> response/"):
            if current_report:
                reports.append(current_report)  # Save the previous report
            current_report = {}
            match = re.search(r"response/(.+)\.md", line)
            if match:
                current_report['asset_name'] = match.group(1)
            else:
                current_report['asset_name'] = "UNKNOWN"  # Handle cases without .md
        elif current_report:
            if "Profit Probability:" in line:
                match = re.search(r"(\d+)%", line)
                if match:
                    current_report['probability'] = int(match.group(1))
                else:
                    current_report['probability'] = 0  # Default if no % found
            # You could add more logic to extract other fields here

    if current_report:  # Add the last report
        reports.append(current_report)

    # Sort by probability (descending)
    sorted_reports = sorted(reports, key=lambda x: x.get('probability', 0), reverse=True)

    for report in sorted_reports:
        print(f"{report['asset_name']}: Probability: {report.get('probability', 'N/A')}%")
