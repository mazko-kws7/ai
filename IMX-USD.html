def simple_moving_average(data, period): ""Calculates the simple moving average (SMA) for a given period."" return pd.Series(data).rolling(window=period).mean().values

def relative_strength_index(data, period): ""Calculates the Relative Strength Index (RSI)."" delta = np.diff(data) up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = pd.Series(up).rolling(period).mean() roll_down1 = pd.Series(down).abs().rolling(period).mean() RS = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS)) return RSI

def macd(data, slow_period, fast_period, signal_period): ""Calculates the Moving Average Convergence Divergence (MACD)."" ema_fast = pd.Series(data).ewm(span=fast_period, adjust=False).mean() ema_slow = pd.Series(data).ewm(span=slow_period, adjust=False).mean() macd_line = ema_fast - ema_slow signal_line = macd_line.ewm(span=signal_period, adjust=False).mean() return macd_line.values, signal_line.values
