# Calculate daily price changes
daily_df['Price_Change'] = daily_df['Close'].diff()

# Calculate gains and losses
daily_df['Gain'] = daily_df['Price_Change'].apply(lambda x: x if x > 0 else 0)
daily_df['Loss'] = daily_df['Price_Change'].apply(lambda x: abs(x) if x < 0 else 0)

# Calculate average gains and losses over 14 days
window = 14
daily_df['Avg_Gain'] = daily_df['Gain'].rolling(window=window, min_periods=window).mean()
daily_df['Avg_Loss'] = daily_df['Loss'].rolling(window=window, min_periods=window).mean()

# Calculate Relative Strength (RS)
daily_df['RS'] = daily_df['Avg_Gain'] / daily_df['Avg_Loss']

# Calculate RSI
daily_df['RSI'] = 100 - (100 / (1 + daily_df['RS']))

print(daily_df[['Close', 'SMA_10', 'SMA_30', 'RSI']].tail(10))
