def calculate_moving_averages(data, window1, window2): """Calculates short-term and long-term moving averages.

Args:
    data (pd.Series): Closing prices.
    window1 (int):  Window for the short-term moving average.
    window2 (int): Window for the long-term moving average.

Returns:
    tuple: Short-term MA, Long-term MA
"""
short_term_ma = data.rolling(window=window1).mean()
long_term_ma = data.rolling(window=window2).mean()
return short_term_ma, long_term_ma

def calculate_rsi(data, window): """Calculates the Relative Strength Index (RSI).

Args:
    data (pd.Series): Closing prices.
    window (int): Window for RSI calculation.

Returns:
    pd.Series: RSI values.
"""
delta = data.diff()
up, down = delta.copy(), delta.copy()
up[up < 0] = 0
down[down > 0] = 0
roll_up = up.rolling(window=window).mean()
roll_down = down.abs().rolling(window=window).mean()

rs = roll_up / roll_down
rsi = 100.0 - (100.0 / (1.0 + rs))
return rsi
