def calculate_rsi(data, period=14): delta = data[‘Close’].diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0

avg_gain = up.rolling(window=period, min_periods=period).mean()
avg_loss = abs(down.rolling(window=period, min_periods=period).mean())

rs = avg_gain / avg_loss
rsi = 100.0 - (100.0 / (1.0 + rs))
return rsi

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9): ema_fast = data[‘Close’].ewm(span=fast_period, adjust=False).mean() ema_slow = data[‘Close’].ewm(span=slow_period, adjust=False).mean() macd = ema_fast - ema_slow signal = macd.ewm(span=signal_period, adjust=False).mean() histogram = macd - signal return macd, signal, histogram
