def calculate_rsi(prices, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given list of prices.

    Args:
        prices (list): A list of closing prices.
        period (int): The period for calculating RSI (default: 14).

    Returns:
        list: A list of RSI values.
    """
    if len(prices) < period:
        return []

    gains = []
    losses = []

    for i in range(1, len(prices)):
        change = prices[i] - prices[i - 1]
        if change > 0:
            gains.append(change)
            losses.append(0)
        else:
            gains.append(0)
            losses.append(abs(change))

    avg_gain = sum(gains[:period]) / period
    avg_loss = sum(losses[:period]) / period

    rsi_values = []
    for i in range(period, len(prices)):
        avg_gain = ((avg_gain * (period - 1)) + gains[i - 1]) / period
        avg_loss = ((avg_loss * (period - 1)) + losses[i - 1]) / period

        if avg_loss == 0:
            rsi = 100
        else:
            rs = avg_gain / avg_loss
            rsi = 100 - (100 / (1 + rs))
        rsi_values.append(rsi)

    return rsi_values

import pandas as pd

def calculate_macd(prices, fast_period=12, slow_period=26, signal_period=9):
    """
    Calculates the Moving Average Convergence Divergence (MACD) for a given list of prices.

    Args:
        prices (list): A list of closing prices.
        fast_period (int): The period for the fast EMA (default: 12).
        slow_period (int): The period for the slow EMA (default: 26).
        signal_period (int): The period for the signal line EMA (default: 9).

    Returns:
        tuple: A tuple containing lists of MACD values, signal line values, and histogram values.
    """
    df = pd.DataFrame(prices, columns=['Close'])

    # Calculate fast EMA
    ema_fast = df['Close'].ewm(span=fast_period, adjust=False).mean()

    # Calculate slow EMA
    ema_slow = df['Close'].ewm(span=slow_period, adjust=False).mean()

    # Calculate MACD line
    macd_line = ema_fast - ema_slow

    # Calculate signal line
    signal_line = macd_line.ewm(span=signal_period, adjust=False).mean()

    # Calculate MACD histogram
    macd_histogram = macd_line - signal_line

    return macd_line.tolist(), signal_line.tolist(), macd_histogram.tolist()
