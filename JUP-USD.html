import pandas as pd
import numpy as np

def calculate_rsi(data, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given dataset.

    Args:
        data (pd.Series): Closing prices.
        period (int): Period for calculating RSI (default is 14).

    Returns:
        pd.Series: RSI values.
    """
    delta = data.diff()
    up = delta.clip(lower=0)
    down = -1 * delta.clip(upper=0)
    
    avg_up = up.rolling(window=period, min_periods=period).mean()
    avg_down = down.rolling(window=period, min_periods=period).mean()

    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    return rsi

def calculate_macd(data, short_window=12, long_window=26, signal_window=9):
    """
    Calculates the Moving Average Convergence Divergence (MACD) for a given dataset.

    Args:
        data (pd.Series): Closing prices.
        short_window (int): Short window for EMA calculation (default is 12).
        long_window (int): Long window for EMA calculation (default is 26).
        signal_window (int): Window for signal line calculation (default is 9).

    Returns:
        tuple: MACD line, signal line, and histogram.
    """
    ema_short = data.ewm(span=short_window, adjust=False).mean()
    ema_long = data.ewm(span=long_window, adjust=False).mean()
    macd_line = ema_short - ema_long
    signal_line = macd_line.ewm(span=signal_window, adjust=False).mean()
    histogram = macd_line - signal_line
    return macd_line, signal_line, histogram

def analyze_ohlcv(ohlcv_data):
    """
    Analyzes OHLCV data using RSI and MACD.

    Args:
        ohlcv_data (pd.DataFrame): DataFrame with 'Open', 'High', 'Low', 'Close', 'Volume' columns.

    Returns:
        tuple: RSI values, MACD line, signal line, and histogram.
    """
    close_prices = ohlcv_data['Close']
    rsi = calculate_rsi(close_prices)
    macd_line, signal_line, histogram = calculate_macd(close_prices)
    return rsi, macd_line, signal_line, histogram

def generate_signals(rsi, macd_line, signal_line, histogram):
    """
    Generates buy and sell signals based on RSI and MACD.

    Args:
        rsi (pd.Series): RSI values.
        macd_line (pd.Series): MACD line.
        signal_line (pd.Series): Signal line.
        histogram (pd.Series): MACD histogram.

    Returns:
        list: List of trading signals (Buy/Sell).
    """
    signals = []
    # RSI oversold/overbought levels
    oversold = 30
    overbought = 70

    # MACD crossover
    macd_crossover_above = (macd_line > signal_line) & (macd_line.shift(1) <= signal_line.shift(1))
    macd_crossover_below = (macd_line < signal_line) & (macd_line.shift(1) >= signal_line.shift(1))

    for i in range(1, len(rsi)):
        if rsi[i] < oversold and macd_crossover_above[i]:
            signals.append("Buy")
        elif rsi[i] > overbought and macd_crossover_below[i]:
            signals.append("Sell")
        else:
            signals.append("Hold")
    return signals
