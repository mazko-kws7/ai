# Calculate RSI (Relative Strength Index) with a 14-day period
def calculate_rsi(data, period=14):
    delta = data.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = up.rolling(window=period, min_periods=period).mean()
    avg_loss = abs(down.rolling(window=period, min_periods=period).mean())
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

df['RSI'] = calculate_rsi(df['Close'])

# Calculate MACD (Moving Average Convergence Divergence)
def calculate_macd(data, short_window=12, long_window=26, signal_window=9):
    # Calculate Short-term EMA
    short_ema = data.ewm(span=short_window, adjust=False).mean()
    
    # Calculate Long-term EMA
    long_ema = data.ewm(span=long_window, adjust=False).mean()
    
    # Calculate MACD Line
    macd_line = short_ema - long_ema
    
    # Calculate Signal Line
    signal_line = macd_line.ewm(span=signal_window, adjust=False).mean()
    
    # Calculate MACD Histogram
    macd_histogram = macd_line - signal_line
    
    return macd_line, signal_line, macd_histogram

macd_line, signal_line, macd_histogram = calculate_macd(df['Close'])
