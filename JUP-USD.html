def calculate_rsi(data, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given dataset.

    Args:
        data (list): List of closing prices.
        period (int): Number of periods to consider for RSI calculation.

    Returns:
        list: List of RSI values. Returns an empty list if the data is insufficient.
    """
    if len(data) < period + 1:
        print("Not enough data to calculate RSI.")
        return []

    # 1. Calculate Daily Price Changes
    price_changes = [data[i] - data[i - 1] for i in range(1, len(data))]

    # 2. Separate Gains and Losses
    gains = [change if change > 0 else 0 for change in price_changes]
    losses = [-change if change < 0 else 0 for change in price_changes]

    # 3. Calculate Average Gain and Average Loss (First values)
    avg_gain = sum(gains[:period]) / period
    avg_loss = sum(losses[:period]) / period

    # Initialize RSI values list
    rsi_values = [0] * len(data)
    # RSI is not defined before the initial period, so pad with 0
    rsi_values[:period] = [0] * period

    # Calculate Relative Strength (RS) and RSI
    if avg_loss == 0:  # Avoid division by zero
        rs = float('inf')  # or some large number
    else:
        rs = avg_gain / avg_loss

    rsi = 100 - (100 / (1 + rs))
    rsi_values[period] = rsi

    # Subsequent average gains and losses and RSI calculations
    for i in range(period + 1, len(data)):
        # Subsequent average gains and losses are calculated by smoothing the previous averages with the current gain or loss
        avg_gain = (avg_gain * (period - 1) + gains[i - 1]) / period
        avg_loss = (avg_loss * (period - 1) + losses[i - 1]) / period

        if avg_loss == 0:  # Avoid division by zero
            rs = float('inf')  # or some large number
        else:
            rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        rsi_values[i] = rsi

    return rsi_values
