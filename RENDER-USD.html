def calculate_rsi(data, window=14): delta = data[‘Close’].diff() up = delta.clip(lower=0) down = -1 * delta.clip(upper=0)

avg_up = up.rolling(window=window, min_periods=window).mean()
avg_down = down.rolling(window=window, min_periods=window).mean()

rs = avg_up / avg_down
rsi = 100 - (100 / (1 + rs))
return rsi
