def calculate_sma(data, period):
    sma_values = []
    for i in range(len(data)):
        if i < period - 1:
            sma_values.append(None)  # SMA not available for the first 'period' days
        else:
            window = data[i - period + 1:i + 1]
            sma = sum(window) / period
            sma_values.append(sma)
    return sma_values

def calculate_rsi(closes, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given list of closing prices.

    Args:
      closes (list): A list of closing prices.
      period (int): The period over which to calculate the RSI (default is 14).

    Returns:
      list: A list of RSI values. Returns None for the first 'period' values where RSI cannot be calculated.
    """
    if len(closes) < period:
        return [None] * len(closes)  # Not enough data to calculate RSI

    gains = [0.0]  # List to hold upward price changes
    losses = [0.0] # List to hold downward price changes

    for i in range(1, len(closes)):
        change = closes[i] - closes[i-1]
        if change > 0:
            gains.append(change)
            losses.append(0.0)
        else:
            losses.append(abs(change))
            gains.append(0.0)

    avg_gain = sum(gains[1:period+1]) / period  # Average gain over the period
    avg_loss = sum(losses[1:period+1]) / period  # Average loss over the period

    rs_values = [None] * period  # Initialize RSI list with None values
    rsi_values = [None] * period

    # Calculate RS and RSI for the first time after 'period'
    if avg_loss != 0:
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
    else:
        rsi = 100 if avg_gain > 0 else 0  # If avg_loss is zero

    rsi_values.append(rsi)
    rs_values.append(rs)

    # Calculate RS and RSI for the rest of the data using the Wilder's smoothing method
    for i in range(period + 1, len(closes)):
        avg_gain = (avg_gain * (period - 1) + gains[i]) / period
        avg_loss = (avg_loss * (period - 1) + losses[i]) / period

        if avg_loss != 0:
            rs = avg_gain / avg_loss
            rsi = 100 - (100 / (1 + rs))
        else:
            rsi = 100 if avg_gain > 0 else 0  # If avg_loss is zero

        rsi_values.append(rsi)
        rs_values.append(rs)

    return rsi_values
