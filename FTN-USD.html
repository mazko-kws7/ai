import numpy as np
import pandas as pd

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = data['Close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    down = abs(down)
    roll_up = up.rolling(period).mean()
    roll_down = down.rolling(period).mean()
    rs = roll_up / roll_down
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_sma(data, period=20):
    """Calculates the Simple Moving Average (SMA)."""
    return data['Close'].rolling(window=period).mean()

def calculate_ema(data, period=20):
    """Calculates the Exponential Moving Average (EMA)."""
    return data['Close'].ewm(span=period, adjust=False).mean()

def assess_trend(data, sma_period=20):
    """
    Assess the trend based on the SMA.

    Returns:
    - "Uptrend" if the price is consistently above the SMA.
    - "Downtrend" if the price is consistently below the SMA.
    - "Sideways" if the price fluctuates around the SMA.
    """
    sma = calculate_sma(data, sma_period)
    last_price = data['Close'].iloc[-1]
    last_sma = sma.iloc[-1]

    if last_price > last_sma:
        return "Uptrend"
    elif last_price < last_sma:
        return "Downtrend"
    else:
        return "Sideways"

def analyze_volume(data):
    """Analyzes volume trends. Returns 'Increasing', 'Decreasing', or 'Stable'."""
    volume_change = data['Volume'].pct_change()
    if volume_change.iloc[-1] > 0.1:  # Significant increase
        return 'Increasing'
    elif volume_change.iloc[-1] < -0.1: # Significant decrease
        return 'Decreasing'
    else:
        return 'Stable'
