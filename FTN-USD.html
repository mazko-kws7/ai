def calculate_sma(data, period): ""Calculates the Simple Moving Average (SMA)."" return pd.Series(data).rolling(window=period).mean().iloc[-1]

def calculate_rsi(data, period=14): ""Calculates the Relative Strength Index (RSI)."" delta = pd.Series(data).diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = up.ewm(span=period, adjust=False).mean() roll_down1 = down.abs().ewm(span=period, adjust=False).mean() RS = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS)) return RSI.iloc[-1]

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9): ""Calculates the Moving Average Convergence Divergence (MACD)."" ema_fast = pd.Series(data).ewm(span=fast_period, adjust=False).mean() ema_slow = pd.Series(data).ewm(span=slow_period, adjust=False).mean() macd = ema_fast - ema_slow signal = macd.ewm(span=signal_period, adjust=False).mean() return macd.iloc[-1], signal.iloc[-1]
