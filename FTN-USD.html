def calculate_rsi(data, window=14): delta = data.diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = up.rolling(window).mean() roll_down1 = down.abs().rolling(window).mean() RS1 = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS1)) return RSI

def calculate_macd(data, fast=12, slow=26, signal=9): EMA_fast = data.ewm(span=fast, min_periods=fast).mean() EMA_slow = data.ewm(span=slow, min_periods=slow).mean() MACD = EMA_fast - EMA_slow MACD_signal = MACD.ewm(span=signal, min_periods=signal).mean() return MACD, MACD_signal
