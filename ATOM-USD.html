def calculate_sma(data, period):
    """Calculates the Simple Moving Average (SMA) for a given period."""
    if len(data) < period:
        return None
    closes = [float(row.split(',')[4]) for row in data[-period:]]  # Extract closing prices
    return sum(closes) / period
