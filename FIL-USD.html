def calculate_rsi(data, period=14): ""“Calculates the Relative Strength Index (RSI).”"" delta = data[‘Close’].diff() up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = up.rolling(period).mean() roll_down1 = down.abs().rolling(period).mean() RS = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS)) return RSI
