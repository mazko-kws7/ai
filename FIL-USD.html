import numpy as np

def calculate_rsi(closing_prices, period=14):
    """
    Calculates the Relative Strength Index (RSI) for a given list of closing prices.

    Args:
        closing_prices (list): A list of closing prices.
        period (int): The period over which to calculate the RSI (default: 14).

    Returns:
        float: The RSI value, or None if there is not enough data.
    """
    if len(closing_prices) < period + 1:
        return None

    # Calculate price changes
    price_changes = np.diff(closing_prices)

    # Split into gains and losses
    gains = price_changes[price_changes > 0]
    losses = -price_changes[price_changes < 0]

    # Calculate average gain and average loss over the period
    avg_gain = np.mean(gains[-period:]) if len(gains) > 0 else 0
    avg_loss = np.mean(losses[-period:]) if len(losses) > 0 else 0

    # Calculate relative strength
    if avg_loss == 0:
        return 100  # RSI is 100 if there are only gains
    relative_strength = avg_gain / avg_loss

    # Calculate RSI
    rsi = 100 - (100 / (1 + relative_strength))

    return rsi

rsi = calculate_rsi(closing_prices)
