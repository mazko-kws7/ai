# Calculate 14-day RSI
def calculate_rsi(data, period=14):
    delta = data.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    avg_gain = up.rolling(window=period, min_periods=period).mean()
    avg_loss = abs(down.rolling(window=period, min_periods=period).mean())
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

df_daily['RSI'] = calculate_rsi(df_daily['Close'])

# Calculate MACD (12-day EMA, 26-day EMA, 9-day SMA of difference)
def calculate_macd(data, period_long=26, period_short=12, period_signal=9):
    ema_long = data.ewm(span=period_long, adjust=False).mean()
    ema_short = data.ewm(span=period_short, adjust=False).mean()
    macd = ema_short - ema_long
    signal = macd.ewm(span=period_signal, adjust=False).mean()
    histogram = macd - signal
    return macd, signal, histogram
