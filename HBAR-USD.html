import numpy as np

def calculate_sma(data, period):
    if len(data) < period:
        return None
    return np.mean(data[-period:])

def calculate_ema(data, period, smoothing=2):
    if len(data) < period:
        return None
    ema = [np.mean(data[:period])]
    for i in range(period, len(data)):
        ema.append((data[i] * (smoothing / (1 + period))) + ema[-1] * (1 - (smoothing / (1 + period))))
    return ema[-1]
