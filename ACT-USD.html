def analyze_daily_data(daily_data): """Analyzes daily OHLCV data to identify trends and potential signals.

Args:
    daily_data (pd.DataFrame): DataFrame with Open, High, Low, Close, Volume data.

Returns:
    dict: Analysis results including trends, volatility, and potential signals.
"""
df = pd.DataFrame(daily_data)
df.columns = ['Open', 'High', 'Low', 'Close', 'Volume']

# Calculate Moving Averages
df['MA_20'] = df['Close'].rolling(window=20).mean()
df['MA_50'] = df['Close'].rolling(window=50).mean()

# Calculate RSI (Relative Strength Index)
delta = df['Close'].diff()
up, down = delta.clip(lower=0), -delta.clip(upper=0)
alpha = 14  # Smoothing factor for RSI
roll_up1 = up.ewm(alpha=1/alpha).mean()
roll_down1 = down.ewm(alpha=1/alpha).mean()
RS1 = roll_up1 / roll_down1
df['RSI'] = 100.0 - (100.0 / (1.0 + RS1))

# Calculate MACD (Moving Average Convergence Divergence)
exp12 = df['Close'].ewm(span=12, adjust=False).mean()
exp26 = df['Close'].ewm(span=26, adjust=False).mean()
macd = exp12 - exp26
signal = macd.ewm(span=9, adjust=False).mean()
df['MACD'] = macd
df['Signal'] = signal
df['MACD_Diff'] = df['MACD'] - df['Signal']

# Identify Trends
if df['MA_20'][-1] > df['MA_50'][-1]:
    trend = "Uptrend"
else:
    trend = "Downtrend"

# Identify potential buy/sell signals based on Moving Average crossovers, RSI, and MACD
signal = "Neutral"
if (df['MA_20'][-2] < df['MA_50'][-2]) and (df['MA_20'][-1] > df['MA_50'][-1]):
    signal = "Buy" # 20-day MA crosses above 50-day MA
elif (df['MA_20'][-2] > df['MA_50'][-2]) and (df['MA_20'][-1] < df['MA_50'][-1]):
    signal = "Sell" # 20-day MA crosses below 50-day MA
elif df['RSI'][-1] < 30:
    signal = "Buy" # Oversold condition
elif df['RSI'][-1] > 70:
    signal = "Sell" # Overbought condition
elif df['MACD_Diff'][-2] < 0 and df['MACD_Diff'][-1] > 0:
    signal = "Buy"  # MACD crossover
elif df['MACD_Diff'][-2] > 0 and df['MACD_Diff'][-1] < 0:
    signal = "Sell" # MACD crossover

analysis = {
    "Trend": trend,
    "Volatility": df['Close'].std(),
    "Signal": signal,
    "RSI": df['RSI'][-1],
    "MACD_Diff": df['MACD_Diff'][-1],
    "MA_20": df['MA_20'][-1],
    "MA_50": df['MA_50'][-1]
}
return analysis
