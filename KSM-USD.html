def analyze_ohlcv(daily_data, hourly_data):

# Daily Data Analysis
daily_df = pd.read_csv(StringIO(daily_data))

daily_df['SMA_20'] = daily_df['Close'].rolling(window=20).mean()
daily_df['SMA_50'] = daily_df['Close'].rolling(window=50).mean()

# Hourly Data Analysis
hourly_df = pd.read_csv(StringIO(hourly_data))
hourly_df.columns = ['Open', 'High', 'Low', 'Close', 'Volume']

# Calculate SMAs for Hourly Data
hourly_df['SMA_20'] = hourly_df['Close'].rolling(window=20).mean()

stop_loss = None

    signal = "Short"

    probability = "Medium-High" #Higher SMA cross probability make profit

    signal = "Long"

    risk_reward_ratio = 2
    probability = "Medium-High" #Higher SMA cross probability make profit
else:
    signal = "Neutral"
    position = "Neutral"

    'signal':signal,

    'stop_loss': stop_loss,
    'take_profit': take_profit,
    'probability':probability,

}
