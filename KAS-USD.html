def calculate_rsi(prices, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    deltas = [prices[i+1] - prices[i] for i in range(len(prices)-1)]
    seed = deltas[:period]
    up = sum([x for x in seed if x > 0]) / period
    down = -sum([x for x in seed if x < 0]) / period
    rs = up / down
    rsi = 100 - (100 / (1 + rs))

    for i in range(period, len(deltas)):
        delta = deltas[i]
        if delta > 0:
            up = (up * (period - 1) + delta) / period
            down = (down * (period - 1)) / period
        else:
            up = (up * (period - 1)) / period
            down = (down * (period - 1) - delta) / period
        rs = up / down
        rsi = 100 - (100 / (1 + rs))
    return rsi


def calculate_macd(prices, slow_period=26, fast_period=12):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    ema_slow = sum(prices[:slow_period]) / slow_period
    k_slow = 2 / (slow_period + 1)
    for i in range(slow_period, len(prices)):
        ema_slow = (prices[i] * k_slow) + (ema_slow * (1 - k_slow))

    ema_fast = sum(prices[:fast_period]) / fast_period
    k_fast = 2 / (fast_period + 1)
    for i in range(fast_period, len(prices)):
        ema_fast = (prices[i] * k_fast) + (ema_fast * (1 - k_fast))
    macd = ema_fast - ema_slow
    return macd
