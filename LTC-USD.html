def calculate_sma(data, period):
    """Calculates the Simple Moving Average (SMA) for a given period."""
    if len(data) < period:
        return None  # Not enough data

    closing_prices = [float(row.split(',')[4]) for row in data]
    sma = sum(closing_prices[-period:]) / period
    return sma
