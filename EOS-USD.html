def calculate_sma(data, period): ""Calculates the Simple Moving Average (SMA)."" return pd.Series(data).rolling(window=period).mean().fillna(0).tolist()

def calculate_rsi(data, period=14): ""Calculates the Relative Strength Index (RSI)."" delta = np.diff(data) up, down = delta.copy(), delta.copy() up[up < 0] = 0 down[down > 0] = 0 roll_up1 = pd.Series(up).rolling(period).mean() roll_down1 = pd.Series(down).abs().rolling(period).mean() RS = roll_up1 / roll_down1 RSI = 100.0 - (100.0 / (1.0 + RS)) return RSI.fillna(0).tolist() # Fill NaN values with 0

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9): ""Calculates the Moving Average Convergence Divergence (MACD)."" ema_fast = pd.Series(data).ewm(span=fast_period, adjust=False).mean() ema_slow = pd.Series(data).ewm(span=slow_period, adjust=False).mean() macd_line = ema_fast - ema_slow signal_line = macd_line.ewm(span=signal_period, adjust=False).mean() histogram = macd_line - signal_line return macd_line.fillna(0).tolist(), signal_line.fillna(0).tolist(), histogram.fillna(0).tolist()

def find_local_minima(data, order=5): ""Finds local minima in the data."" min_idx = argrelextrema(np.array(data), np.less, order=order)[0] return min_idx

def find_local_maxima(data, order=5): ""Finds local maxima in the data."" max_idx = argrelextrema(np.array(data), np.greater, order=order)[0] return max_idx
