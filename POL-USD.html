# Calculate gains and losses

df['Loss'] = df['Price Change'].apply(lambda x: abs(x) if x < 0 else 0)

# Calculate average gains and losses over the last 14 days
avg_gain = df['Gain'].rolling(window=14, min_periods=14).mean().iloc[-1]
avg_loss = df['Loss'].rolling(window=14, min_periods=14).mean().iloc[-1]

# Calculate the Relative Strength (RS)
rs = avg_gain / avg_loss if avg_loss != 0 else 0

# Calculate the RSI
rsi = 100 - (100 / (1 + rs))
