import pandas as pd
import numpy as np

def calculate_sma(data, period):
  """Calculates the Simple Moving Average (SMA)."""
  sma = pd.Series(data).rolling(window=period).mean().fillna(0).tolist()
  return sma

def calculate_rsi(prices, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    deltas = np.diff(prices)
    seed = deltas[:period+1]
    up = seed[seed>=0].sum()/period
    down = -seed[seed<0].sum()/period
    rs = up/down
    rsi = np.zeros_like(prices)
    rsi[:period] = 100. - 100./(1.+rs)

    for i in range(period, len(prices)):
        delta = deltas[i-1] # cause the diff is 1 shorter

        if delta>0:
            upval = delta
            downval = 0.
        else:
            upval = 0.
            downval = -delta

        up = (up*(period-1) + upval)/period
        down = (down*(period-1) + downval)/period

        rs = up/down
        rsi[i] = 100. - 100./(1.+rs)

    return rsi.tolist()


def calculate_macd(prices, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    ema_fast = pd.Series(prices).ewm(span=fast_period, adjust=False).mean()
    ema_slow = pd.Series(prices).ewm(span=slow_period, adjust=False).mean()
    macd_line = ema_fast - ema_slow
    signal_line = pd.Series(macd_line).ewm(span=signal_period, adjust=False).mean().fillna(0).tolist()
    macd_line = macd_line.fillna(0).tolist()  # Fill NaN values with 0 after conversion to series
    histogram = (pd.Series(macd_line) - pd.Series(signal_line)).tolist()
    return macd_line, signal_line, histogram
