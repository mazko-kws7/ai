import pandas as pd
import numpy as np

def calculate_rsi(data, period=14):
    """Calculates the Relative Strength Index (RSI)."""
    delta = data['Close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    down = abs(down)
    
    roll_up = up.rolling(period).mean()
    roll_down = down.rolling(period).mean()

    rs = roll_up / roll_down
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi

def calculate_macd(data, fast_period=12, slow_period=26, signal_period=9):
    """Calculates the Moving Average Convergence Divergence (MACD)."""
    ema_fast = data['Close'].ewm(span=fast_period, adjust=False).mean()
    ema_slow = data['Close'].ewm(span=slow_period, adjust=False).mean()
    macd = ema_fast - ema_slow
    signal = macd.ewm(span=signal_period, adjust=False).mean()
    histogram = macd - signal
    return macd, signal, histogram

def analyze_price_action(data):
  """
  Analyzes price action for potential patterns (e.g., bullish/bearish engulfing, doji).
  This is a simplified example and can be expanded with more sophisticated pattern recognition.
  """
  # Bullish Engulfing: A small bearish candle followed by a large bullish candle that engulfs the previous candle.
  # Bearish Engulfing: A small bullish candle followed by a large bearish candle that engulfs the previous candle.
  # Doji: A candle with a very small body, indicating indecision.
  signals = []
  for i in range(1, len(data)):
    prev_open = data['Open'][i-1]
    prev_close = data['Close'][i-1]
    curr_open = data['Open'][i]
    curr_close = data['Close'][i]

    if (prev_close < prev_open) and (curr_close > curr_open) and (curr_close > prev_open) and (curr_open < prev_close):
      signals.append({'index': i, 'type': 'Bullish Engulfing'})
    elif (prev_close > prev_open) and (curr_close < curr_open) and (curr_close < prev_open) and (curr_open > prev_close):
      signals.append({'index': i, 'type': 'Bearish Engulfing'})
    elif abs(curr_close - curr_open) < (data['High'][i] - data['Low'][i]) * 0.1: # Body is 10% of the range
      signals.append({'index': i, 'type': 'Doji'})
  return signals

def generate_signal(signal_type, confidence, justification, relevant_data, recommended_action, take_profit_probability):
    """Generates a trading signal in the specified format."""
    return {
        'Signal Type': signal_type,
        'Cryptocurrency': 'POL-USD',
        'Confidence Level': confidence,
        'Take Profit Probability': take_profit_probability,
        'Justification': justification,
        'Relevant Data': relevant_data,
        'Recommended Action': recommended_action
    }
