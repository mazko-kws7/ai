import numpy as np
import pandas as pd

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    return pd.Series(data).rolling(window=period).mean().iloc[-1]

def calculate_rsi(data, period=14):
    """Calculates Relative Strength Index."""
    delta = np.diff(data)
    up, down = delta.copy(), delta.copy()
    up[up<0] = 0
    down[down>0] = 0

    avg_up = pd.Series(up).rolling(window=period,min_periods=period).mean()
    avg_down = abs(pd.Series(down).rolling(window=period,min_periods=period).mean())
    RS = avg_up / avg_down
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI.iloc[-1]

def assess_trend(data, period=20):
    """Simple trend assessment based on moving average."""
    current_price = data[-1]
    moving_average = calculate_sma(data, period)
    if current_price > moving_average:
        return "Uptrend"
    elif current_price < moving_average:
        return "Downtrend"
    else:
        return "Sideways"

def analyze_volume(volume_data):
    """Analyzes volume changes."""
    if len(volume_data) < 2:
        return "Insufficient volume data"
    recent_volume = volume_data[-1]
    previous_volume = volume_data[-2]
    volume_change = recent_volume - previous_volume
    if volume_change > 0:
        return "Volume increasing"
    elif volume_change < 0:
        return "Volume decreasing"
    else:
        return "Volume stable"
