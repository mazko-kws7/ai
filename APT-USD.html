def calculate_sma(data, period=20): return data[Close].rolling(window=period).mean()

def calculate_rsi(data, period=14): delta = data[Close].diff() up = delta.clip(lower=0) down = -1 * delta.clip(upper=0)

# Calculate Exponential Moving Averages (EMAs) for up and down
ema_up = up.ewm(com=period - 1, adjust=False).mean()
ema_down = down.ewm(com=period - 1, adjust=False).mean()

rs = ema_up / ema_down
rsi = 100 - (100 / (1 + rs))
return rsi
