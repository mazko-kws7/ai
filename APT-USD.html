def calculate_sma(data, period):
    closes = np.array([float(row[4]) for row in data])
    if len(closes) < period:
        return None
    sma = np.mean(closes[-period:])
    return sma

# Relative Strength Index (RSI) - Daily Data
def calculate_rsi(data, period=14):
    closes = np.array([float(row[4]) for row in data])
    if len(closes) < period:
        return None

    deltas = np.diff(closes)
    seed = deltas[:period]
    up = seed[seed >= 0].sum() / period
    down = -seed[seed < 0].sum() / period
    rs = up / down if down != 0 else 0
    rsi = 100. - 100. / (1. + rs)

    upward_changes = deltas[deltas >= 0]
    downward_changes = -deltas[deltas < 0]

    up = np.mean(upward_changes)
    down = np.mean(downward_changes) if len(downward_changes) > 0 else 0
    rs = up / down if down != 0 else 0
    rsi = 100. - 100. / (1. + rs)
    return rsi
