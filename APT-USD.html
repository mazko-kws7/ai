import numpy as np
import pandas as pd

def calculate_sma(data, period):
    """Calculates Simple Moving Average."""
    return pd.Series(data).rolling(window=period).mean().iloc[-1]

def calculate_rsi(data, period):
    """Calculates Relative Strength Index."""
    delta = pd.Series(data).diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up1 = up.ewm(span=period).mean()
    roll_down1 = down.abs().ewm(span=period).mean()
    RS = roll_up1 / roll_down1
    RSI = 100.0 - (100.0 / (1.0 + RS))
    return RSI.iloc[-1]

def calculate_macd(data, fast_period, slow_period, signal_period):
    """Calculates Moving Average Convergence Divergence."""
    fast_ema = pd.Series(data).ewm(span=fast_period).mean()
    slow_ema = pd.Series(data).ewm(span=slow_period).mean()
    macd = fast_ema - slow_ema
    signal = macd.ewm(span=signal_period).mean()
    return macd.iloc[-1], signal.iloc[-1]

def calculate_bollinger_bands(data, period, num_std_dev):
    """Calculates Bollinger Bands."""
    mean = pd.Series(data).rolling(window=period).mean().iloc[-1]
    std_dev = pd.Series(data).rolling(window=period).std().iloc[-1]
    upper_band = mean + num_std_dev * std_dev
    lower_band = mean - num_std_dev * std_dev
    return upper_band, mean, lower_band
